מה בדף הזה?
הדף מסיים את אגדות משה: קברו הנסתר, הליכה בדרכי ה׳, ומדוע נתאווה להיכנס לארץ. פרק ג מתחיל: מנחת הסוטה — סדר הקרבתה, כלי כסף וזהב מול כפיפה מצרית, שלבי ההקרבה, קמיצה, קידוש כפול ולבונה.
גַּסְטְרָא שֶׁל בֵּית פְּעוֹר: הַרְאֵנוּ הֵיכָן מֹשֶׁה קָבוּר. עָמְדוּ לְמַעְלָה — נִדְמָה לָהֶם לְמַטָּה, לְמַטָּה — נִדְמָה לָהֶם לְמַעְלָה. נֶחְלְקוּ לִשְׁתֵּי כִיתּוֹת, אוֹתָן שֶׁעוֹמְדִים לְמַעְלָה — נִדְמָה לָהֶן לְמַטָּה, לְמַטָּה — נִדְמָה לָהֶן לְמַעְלָה, לְקַיֵּים מַה שֶּׁנֶּאֱמַר ״וְלֹא יָדַע אִישׁ אֶת קְבֻרָתוֹ״.
הרומאים שלחו חיילים לבית פעור לחפש את קבר משה. עמדו למעלה ונדמה להם שהקבר למטה, עמדו למטה ונדמה למעלה. נחלקו לשתי כיתות, וכל כיתה ראתה את הקבר בצד ההפוך, שנאמר "ולא ידע איש את קבורתו".
רַבִּי חָמָא בְּרַבִּי חֲנִינָא אָמַר: אַף מֹשֶׁה רַבֵּינוּ אֵינוֹ יוֹדֵעַ הֵיכָן קָבוּר. כְּתִיב הָכָא: ״וְלֹא יָדַע אִישׁ אֶת קְבֻרָתוֹ״, וּכְתִיב הָתָם: ״וְזֹאת הַבְּרָכָה אֲשֶׁר בֵּרַךְ מֹשֶׁה אִישׁ הָאֱלֹהִים״. וְאָמַר רַבִּי חָמָא בְּרַבִּי חֲנִינָא: מִפְּנֵי מָה נִקְבַּר מֹשֶׁה אֵצֶל בֵּית פְּעוֹר — כְּדֵי לְכַפֵּר עַל מַעֲשֵׂה פְעוֹר.
ומכאן: ר' חמא מפרש שאפילו משה עצמו אינו יודע היכן קבור, שהרי "איש" בפסוק היינו "משה איש האלהים". ומדוע נקבר דווקא ליד בית פעור? כדי לכפר על חטא פעור שחטאו בו ישראל.
וְאָמַר רַבִּי חָמָא בְּרַבִּי חֲנִינָא, מַאי דִּכְתִיב: ״אַחֲרֵי ה׳ אֱלֹהֵיכֶם תֵּלֵכוּ״, וְכִי אֶפְשָׁר לוֹ לְאָדָם לְהַלֵּךְ אַחַר שְׁכִינָה? וַהֲלֹא כְּבָר נֶאֱמַר ״כִּי ה׳ אֱלֹהֶיךָ אֵשׁ אוֹכְלָה הוּא״!
ר' חמא שואל: "אחרי ה' אלהיכם תלכו" — וכי אפשר להלך אחרי השכינה? הרי "ה' אלהיך אש אוכלה"!
אֶלָּא, לְהַלֵּךְ אַחַר מִדּוֹתָיו שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא: מָה הוּא מַלְבִּישׁ עֲרוּמִּים, דִּכְתִיב: ״וַיַּעַשׂ ה׳ אֱלֹהִים לְאָדָם וּלְאִשְׁתּוֹ כׇּתְנוֹת עוֹר וַיַּלְבִּשֵׁם״ — אַף אַתָּה הַלְבֵּשׁ עֲרוּמִּים. הַקָּדוֹשׁ בָּרוּךְ הוּא בִּיקֵּר חוֹלִים, דִּכְתִיב: ״וַיֵּרָא אֵלָיו ה׳ בְּאֵלֹנֵי מַמְרֵא״ — אַף אַתָּה בַּקֵּר חוֹלִים. הַקָּדוֹשׁ בָּרוּךְ הוּא נִיחֵם אֲבֵלִים, דִּכְתִיב: ״וַיְהִי אַחֲרֵי מוֹת אַבְרָהָם וַיְבָרֶךְ אֱלֹהִים אֶת יִצְחָק בְּנוֹ״ — אַף אַתָּה נַחֵם אֲבֵלִים. הַקָּדוֹשׁ בָּרוּךְ הוּא קָבַר מֵתִים, דִּכְתִיב: ״וַיִּקְבֹּר אוֹתוֹ בַּגַּי״ — אַף אַתָּה קְבוֹר מֵתִים.
אלא הכוונה: הלך אחר מידותיו. ① מלביש ערומים, שעשה כתנות עור לאדם וחוה. ② מבקר חולים, שנראה לאברהם אחרי המילה. ③ מנחם אבלים, שברך את יצחק אחרי מות אברהם. ④ קובר מתים, שקבר את משה בעצמו.
״כׇּתְנוֹת עוֹר״. רַב וּשְׁמוּאֵל, חַד אָמַר: דָּבָר הַבָּא מִן הָעוֹר, וְחַד אָמַר: דָּבָר שֶׁהָעוֹר נֶהֱנֶה מִמֶּנּוּ.
ולגבי "כתנות עור" — מחלוקת רב ושמואל: ① דבר הבא מן העור, כלומר עור ממש. ② דבר שהעור נהנה ממנו, כלומר בגד רך ונעים.
דָּרַשׁ רַבִּי שִׂמְלַאי: תּוֹרָה תְּחִלָּתָהּ גְּמִילוּת חֲסָדִים, וְסוֹפָהּ גְּמִילוּת חֲסָדִים. תְּחִילָּתָהּ גְּמִילוּת חֲסָדִים — דִּכְתִיב: ״וַיַּעַשׂ ה׳ אֱלֹהִים לְאָדָם וּלְאִשְׁתּוֹ כׇּתְנוֹת עוֹר וַיַּלְבִּשֵׁם״, וְסוֹפָהּ גְּמִילוּת חֲסָדִים — דִּכְתִיב: ״וַיִּקְבֹּר אוֹתוֹ בַּגַּי״.
ומכאן למד ר' שמלאי: התורה כולה מתחילה בגמילות חסדים (הלבשת אדם וחוה) ומסתיימת בגמילות חסדים (קבורת משה). זהו עיקר התורה, חסד מתחילתה ועד סופה.
דָּרַשׁ רַבִּי שִׂמְלַאי: מִפְּנֵי מָה נִתְאַוָּה מֹשֶׁה רַבֵּינוּ לִיכָּנֵס לְאֶרֶץ יִשְׂרָאֵל? וְכִי לֶאֱכוֹל מִפִּרְיָהּ הוּא צָרִיךְ?! אוֹ לִשְׂבּוֹעַ מִטּוּבָהּ הוּא צָרִיךְ?! אֶלָּא כָּךְ אָמַר מֹשֶׁה: הַרְבֵּה מִצְוֹת נִצְטַוּוּ יִשְׂרָאֵל, וְאֵין מִתְקַיְּימִין אֶלָּא בְּאֶרֶץ יִשְׂרָאֵל. אֶכָּנֵס אֲנִי לָאָרֶץ כְּדֵי שֶׁיִּתְקַיְּימוּ כּוּלָּן עַל יָדִי.
ר' שמלאי דרש: מדוע נתאווה משה כל כך לארץ ישראל? לא לאכול מפריה ולשבוע מטובה, אלא שהרבה מצוות אינן מתקיימות אלא בארץ ורצה לקיימן בעצמו.
אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא: כְּלוּם אַתָּה מְבַקֵּשׁ אֶלָּא לְקַבֵּל שָׂכָר? מַעֲלֶה אֲנִי עָלֶיךָ כְּאִילּוּ עֲשִׂיתָם, שֶׁנֶּאֱמַר: ״לָכֵן אֲחַלֶּק לוֹ בָרַבִּים וְאֶת עֲצוּמִים יְחַלֵּק שָׁלָל תַּחַת אֲשֶׁר הֶעֱרָה לַמָּוֶת נַפְשׁוֹ וְאֶת פּוֹשְׁעִים נִמְנָה וְהוּא חֵטְא רַבִּים נָשָׂא וְלַפֹּשְׁעִים יַפְגִּיעַ״.
ומשיבו הקב"ה: כוונתך רק לשכר? מעלה אני עליך כאילו עשיתם. ראיה: "לכן אחלק לו ברבים ואת עצומים יחלק שלל", כלומר שכרו גדול כשל האבות.
״לָכֵן אֲחַלֶּק לוֹ בָּרַבִּים״. יָכוֹל כָּאַחֲרוֹנִים וְלֹא כָּרִאשׁוֹנִים, תַּלְמוּד לוֹמַר: ״וְאֶת עֲצוּמִים יְחַלֵּק שָׁלָל״, כְּאַבְרָהָם יִצְחָק וְיַעֲקֹב, שֶׁהֵן עֲצוּמִים בַּתּוֹרָה וּבְמִצְוֹת. ״תַּחַת אֲשֶׁר הֶעֱרָה לַמָּוֶת נַפְשׁוֹ״, שֶׁמָּסַר עַצְמוֹ לְמִיתָה, שֶׁנֶּאֱמַר: ״וְאִם אַיִן מְחֵנִי נָא וְגוֹ׳״.
ומכאן: "תחת אשר הערה למות נפשו" פירושו שמסר עצמו למיתה כשאמר "מחני נא מספרך". "ואת פושעים נמנה" פירושו שנמנה עם מתי מדבר שלא נכנסו לארץ.
״וְאֶת פּוֹשְׁעִים נִמְנָה״ — שֶׁנִּמְנָה עִם מֵתֵי מִדְבָּר. ״וְהוּא חֵטְא רַבִּים נָשָׂא״ — שֶׁכִּיפֵּר עַל מַעֲשֵׂה הָעֵגֶל. ״וְלַפֹּשְׁעִים יַפְגִּיעַ״ — שֶׁבִּיקֵּשׁ רַחֲמִים עַל פּוֹשְׁעֵי יִשְׂרָאֵל שֶׁיַּחְזְרוּ בִּתְשׁוּבָה. וְאֵין פְּגִיעָה אֶלָּא תְּפִלָּה, שֶׁנֶּאֱמַר: ״וְאַתָּה אַל תִּתְפַּלֵּל בְּעַד הָעָם הַזֶּה וְאַל תִּשָּׂא בַעֲדָם רִנָּה וּתְפִלָּה וְאַל תִּפְגַּע בִּי״.
ועוד: "והוא חטא רבים נשא" פירושו שכיפר על מעשה העגל. "ולפושעים יפגיע" פירושו שביקש רחמים על פושעי ישראל שיחזרו בתשובה. "פגיעה" היינו תפילה, כמו "אל תפגע בי".
הֲדַרַן עֲלָךְ הַמְקַנֵּא לְאִשְׁתּוֹ
הדרן עלך פרק "המקנא לאשתו" — סיום הפרק השני של מסכת סוטה.
הָיָה מֵבִיא אֶת מִנְחָתָהּ בְּתוֹךְ כְּפִיפָה מִצְרִית וְנוֹתְנָהּ עַל יָדֶיהָ כְּדֵי לְיַגְּעָהּ.
המשנה (פרק ג) שונה: הבעל מביא את מנחת הסוטה בכפיפה מצרית, כלומר סל פשוט מעלי דקל, ונותנה על ידיה כדי ליגעה, שמא תודה לפני שתשתה.
כׇּל הַמְּנָחוֹת תְּחִילָּתָן וְסוֹפָן בִּכְלִי שָׁרֵת, וְזוֹ תְּחִלָּתָהּ בִּכְפִיפָה מִצְרִית, וְסוֹפָהּ בִּכְלִי שָׁרֵת.
ומכאן ההבדלים בין מנחת סוטה לשאר מנחות: ① שאר מנחות מתחילתן בכלי שרת, וזו מתחילה בכפיפה מצרית. ② שאר מנחות טעונות שמן ולבונה, וזו לא. ③ שאר מנחות מחיטים, וזו משעורים. ④ מנחת העומר גם משעורים אך באה גרש, כלומר מובחר, וזו קמח פשוט. רבן גמליאל נימק: מעשיה כבהמה ולכן קרבנה מאכל בהמה.
כׇּל הַמְּנָחוֹת טְעוּנוֹת שֶׁמֶן וּלְבוֹנָה, וְזוֹ אֵינָהּ טְעוּנָה לֹא שֶׁמֶן וְלֹא לְבוֹנָה. כׇּל הַמְּנָחוֹת בָּאוֹת מִן הַחִטִּין, וְזוֹ בָּאָה מִן הַשְּׂעוֹרִין. מִנְחַת הָעוֹמֶר, אַף עַל פִּי שֶׁבָּאָה מִן הַשְּׂעוֹרִין — הִיא הָיְתָה בָּאָה גֶּרֶשׂ, וְזוֹ בָּאָה קֶמַח. רַבָּן גַּמְלִיאֵל אוֹמֵר: כְּשֵׁם שֶׁמַּעֲשֶׂיהָ מַעֲשֵׂה בְּהֵמָה, כָּךְ קׇרְבָּנָהּ מַאֲכַל בְּהֵמָה.
ובגמרא: אבא חנין בשם ר' אליעזר מפרש שמיגעים אותה כדי שתחזור בה ותודה. ואם התורה חסה כך על עוברי רצונו כגון הסוטה, קל וחומר על עושי רצונו.
גְּמָ׳ תַּנְיָא, אַבָּא חָנִין אוֹמֵר מִשּׁוּם רַבִּי אֱלִיעֶזֶר: [וְכׇל כָּךְ לָמָּה —] כְּדֵי לְיַגְּעָהּ, כְּדֵי שֶׁתַּחֲזוֹר בָּהּ. אִם כָּכָה חָסָה תּוֹרָה עַל עוֹבְרֵי רְצוֹנוֹ, קַל וָחוֹמֶר עַל עוֹשֵׂי רְצוֹנוֹ.
ומקשים: אולי מיגעים אותה לא מחמת חסות, אלא כדי שתודה ולא תימחק מגילת ה'? ודוחים: התנא סובר שמשקים אותה קודם ואחר כך מקריבים מנחתה.
וּמִמַּאי מִשּׁוּם דְּחָסָה הוּא? דִּילְמָא כִּי הֵיכִי דְּלֹא תִּימָּחֵק מְגִילָּה! קָסָבַר
ואם כך המגילה כבר נמחקה לפני הנחת המנחה, וההיגוע בכפיפה לא קשור למחיקה. אלא ברור שהטעם הוא חסות התורה על הסוטה, שתודה ולא תמות.
מַשְׁקָהּ, וְאַחַר כָּךְ מַקְרִיב אֶת מִנְחָתָהּ. דְּאִי מִשּׁוּם מְגִילָּה הָא אִימְּחִיקָא לַהּ.
ומקשים: המשנה אמרה שכל המנחות מתחילתן בכלי שרת, אך הברייתא אומרת שמביאים מנחה בקלתות של כסף וזהב, ולא בכלי שרת!
כׇּל הַמְּנָחוֹת וְכוּ׳. וּרְמִינְהוּ: סֵדֶר מְנָחוֹת, כֵּיצַד? אָדָם מֵבִיא מִנְחָה מִתּוֹךְ בֵּיתוֹ בִּקְלָתוֹת שֶׁל כֶּסֶף וְשֶׁל זָהָב, וְנוֹתְנָהּ לְתוֹךְ כְּלִי שָׁרֵת, וּמְקַדְּשָׁהּ בִּכְלִי שָׁרֵת, וְנוֹתֵן עָלֶיהָ שַׁמְנָהּ וּלְבוֹנָתָהּ, וּמוֹלִיכָהּ אֵצֶל כֹּהֵן. וְכֹהֵן מוֹלִיכָהּ אֵצֶל מִזְבֵּחַ, וּמַגִּישָׁהּ בְּקֶרֶן דְּרוֹמִית מַעֲרָבִית כְּנֶגֶד חוּדָּהּ שֶׁל קֶרֶן, וְדַיּוֹ.
ומשיבים: סדר המנחות בברייתא כך הוא: מביא מביתו בקלתות, נותן בכלי שרת ומקדש, נותן שמן ולבונה, מוליך לכהן, כהן מוליך למזבח, ומגיש בקרן דרומית מערבית.
וּמְסַלֵּק אֶת הַלְּבוֹנָה לְצַד אֶחָד, וְקוֹמֵץ מִמָּקוֹם שֶׁנִּתְרַבָּה שַׁמְנָהּ וְנוֹתְנוֹ לְתוֹךְ כְּלִי שָׁרֵת, וּמְקַדְּשׁוֹ בִּכְלִי שָׁרֵת. וּמְלַקֵּט אֶת לְבוֹנָתָהּ וְנוֹתְנָהּ עַל גַּבָּיו, וּמַעֲלֶה אוֹתוֹ לְגַבֵּי מִזְבֵּחַ, וּמַעֲלֵהוּ וּמַקְטִירוֹ בִּכְלִי שָׁרֵת, וּמוֹלְחוֹ, וְנוֹתְנוֹ עַל גַּבֵּי הָאִישִּׁים.
וממשיך: מסלק הלבונה לצד, קומץ ממקום שנתרבה שמנה, נותן הקומץ בכלי שרת ומקדשו שוב. מלקט הלבונה ונותן על הקומץ, מעלה למזבח, מולח, ומניח על האש.
קָרַב הַקּוֹמֶץ שְׁיָרֶיהָ נֶאֱכָלִין, וְרַשָּׁאִין הַכֹּהֲנִים לִיתֵּן לְתוֹכָהּ יַיִן וְשֶׁמֶן וּדְבַשׁ, וְאֵין אֲסוּרִין אֶלָּא מִלְּחַמֵּץ.
ולאחר שקרב הקומץ, השיריים נאכלים לכהנים. רשאים להוסיף יין, שמן ודבש, ורק אסורים מלחמץ.
קָתָנֵי מִיהָא בִּקְלָתוֹת שֶׁל כֶּסֶף וּבִקְלָתוֹת שֶׁל זָהָב! אָמַר רַב פָּפָּא: אֵימָא בְּכֵלִים הָרְאוּיִין לִכְלֵי שָׁרֵת.
ומקשים: הברייתא אומרת "קלתות של כסף ושל זהב" ולא כלי שרת! רב פפא מתרץ: הכוונה לכלים הראויים להיות כלי שרת מפני שהם ממתכות יקרות, אם היה מקדישם.
מִכְּלַל דִּכְפִיפָה מִצְרִית לָא חַזְיָא, כְּמַאן — דְּלָא כְּרַבִּי יוֹסֵי בְּרַבִּי יְהוּדָה, דְּתַנְיָא: כְּלֵי שָׁרֵת שֶׁעֲשָׂאָן שֶׁל עֵץ, רַבִּי פּוֹסֵל וְרַבִּי יוֹסֵי בְּרַבִּי יְהוּדָה מַכְשִׁיר.
ומכאן: כפיפה מצרית אינה ראויה לכלי שרת. ושואלים: האם זה כר' יוסי בר' יהודה שמכשיר כלי עץ? ומשיבים: לא, אפילו הוא מכשיר רק כלי עץ חשובים, לא פחותים כסל דקל.
אֲפִילּוּ תֵּימָא רַבִּי יוֹסֵי בְּרַבִּי יְהוּדָה, אֵימַר דְּאָמַר רַבִּי יוֹסֵי בְּרַבִּי יְהוּדָה בַּחֲשׁוּבִין, בִּפְחוּתִין מִי אָמַר? לֵית לֵיהּ לְרַבִּי יוֹסֵי בְּרַבִּי יְהוּדָה ״הַקְרִיבֵהוּ נָא לְפֶחָתֶךָ״?!
ולפי זה: "הקריבהו נא לפחתך" מלמד שדבר שאין נותנים כמתנה למושל אין מביאים למקדש. סל דקל פשוט בוודאי אינו ראוי לכלי שרת.
וְנוֹתְנָהּ לִכְלֵי שָׁרֵת, וּמְקַדְּשָׁהּ בִּכְלִי שָׁרֵת. שָׁמְעַתְּ מִינַּהּ — כְּלִי שָׁרֵת אֵין מְקַדְּשִׁין אֶלָּא מִדַּעַת? אֵימָא: נוֹתְנָהּ בִּכְלִי שָׁרֵת לְקַדְּשָׁהּ בִּכְלִי שָׁרֵת.
"נותנה בכלי שרת ומקדשה" — ושואלים: האם כלי שרת מקדשים רק מדעת, כלומר בכוונה? ומשיבים: לא בהכרח, אפשר לפרש שנותנה בכלי שרת כדי לקדשה, ואין צורך בכוונה נפרדת.
וְנוֹתֵן עָלֶיהָ שַׁמְנָהּ וּלְבוֹנָתָהּ, שֶׁנֶּאֱמַר: ״וְיָצַק עָלֶיהָ שֶׁמֶן וְנָתַן עָלֶיהָ לְבֹנָה״.
ולאחר מכן נותן שמן ולבונה, שנאמר "ויצק עליה שמן ונתן עליה לבנה".
וּמוֹלִיכָהּ אֵצֶל כֹּהֵן, דִּכְתִיב: ״וֶהֱבִיאָהּ אֶל בְּנֵי אַהֲרֹן וְגוֹ׳״.
ומשם מוליכה אצל כהן, שנאמר "והביאה אל בני אהרן הכהנים".
וְכֹהֵן מוֹלִיכָהּ אֵצֶל מִזְבֵּחַ, דִּכְתִיב: ״וְהִגִּישָׁהּ אֵצֶל הַמִּזְבֵּחַ״.
והכהן מוליכה אצל מזבח, שנאמר "והגישה אצל המזבח". ומגיש בקרן דרומית מערבית כנגד חודה של קרן, ודי בכך.
מַגִּישָׁהּ בְּקֶרֶן דְּרוֹמִית מַעֲרָבִית כְּנֶגֶד חוּדָּהּ שֶׁל קֶרֶן וְדַיּוֹ. מְנָלַן —
ומנין שבקרן דרומית מערבית? "לפני ה'" היינו מערב, "אל פני המזבח" היינו דרום. וכדי לקיים שניהם מגיש בקרן דרומית מערבית.
דִּכְתִיב: ״וְזֹאת תּוֹרַת הַמִּנְחָה הַקְרֵב אוֹתָהּ בְּנֵי אַהֲרֹן לִפְנֵי ה׳ אֶל פְּנֵי הַמִּזְבֵּחַ״, וְתַנְיָא: ״לִפְנֵי ה׳״, יָכוֹל בַּמַּעֲרָב — תַּלְמוּד לוֹמַר: ״אֶל פְּנֵי הַמִּזְבֵּחַ״. אִי אֶל פְּנֵי הַמִּזְבֵּחַ, יָכוֹל בַּדָּרוֹם — תַּלְמוּד לוֹמַר: ״לִפְנֵי ה׳״. הָא כֵּיצַד — מַגִּישָׁהּ בְּקֶרֶן דְּרוֹמִית מַעֲרָבִית כְּנֶגֶד חוּדָּהּ שֶׁל קֶרֶן, וְדַיּוֹ.
ר' אלעזר מפרש: כשיש שני מקראות סותרים, בוחרים את מי שמקיים את עצמו ואת חברו. "אל פני המזבח", כלומר דרום, מקיים גם "לפני ה'", כלומר מערב, כי כל המזבח בצפון העזרה ודרומו מכוון מול ההיכל.
רַבִּי אֶלְעָזָר אוֹמֵר: יָכוֹל יַגִּישֶׁנָּה בְּמַעֲרָבָהּ שֶׁל קֶרֶן אוֹ לִדְרוֹמָהּ שֶׁל קֶרֶן — אָמַרְתָּ: כׇּל מָקוֹם שֶׁאַתָּה מוֹצֵא שְׁנֵי מִקְרָאוֹת, אֶחָד מְקַיֵּים עַצְמוֹ וּמְקַיֵּים דִּבְרֵי חֲבֵירוֹ, וְאֶחָד מְקַיֵּים עַצְמוֹ וּמְבַטֵּל דִּבְרֵי חֲבֵירוֹ — מַנִּיחִין אֶת שֶׁמְּקַיֵּים עַצְמוֹ וּמְבַטֵּל חֲבֵירוֹ, וְתוֹפְסִין אֶת שֶׁמְּקַיֵּים עַצְמוֹ וּמְקַיֵּים חֲבֵירוֹ. כְּשֶׁאַתָּה אוֹמֵר ״לִפְנֵי ה׳״ בַּמַּעֲרָב — בִּטַּלְתָּה ״אֶל פְּנֵי הַמִּזְבֵּחַ״ בַּדָּרוֹם, וּכְשֶׁאַתָּה אוֹמֵר ״אֶל פְּנֵי הַמִּזְבֵּחַ״ בַּדָּרוֹם, קִיַּימְתָּה ״לִפְנֵי ה׳״ בַּמַּעֲרָב. הָא כֵּיצַד? מַגִּישָׁהּ לִדְרוֹמָהּ שֶׁל קֶרֶן.
ומכאן: "ודיו" — מה מוסיף? רב אשי מפרש: שלא תחשוב שצריך לגעת במנחה ממש בקרן המזבח, כלומר הגשה גופנית. קמשמע לן שמספיק להגיש בכלי.
וְהֵיכָן קִיַּימְתָּה? אָמַר רַב אָשֵׁי: קָסָבַר הַאי תַּנָּא כּוּלֵּיהּ מִזְבֵּחַ בְּצָפוֹן קָאֵי.
ומנין שההגשה בכלי? "והקריבה אל הכהן והגישה אל המזבח" מלמד: מה הקרבה אצל כהן בכלי, אף הגשה אצל מזבח בכלי, ולא המנחה בעצמה.
מַאי ״וְדַיּוֹ״? אָמַר רַב אָשֵׁי: אִיצְטְרִיךְ, סָלְקָא דַּעְתָּךְ אָמֵינָא: תִּיבְּעֵי הַגָּשַׁת מִנְחָה גּוּפַהּ, קָמַשְׁמַע לַן.
מסלק את הלבונה לפני הקמיצה, כדי שלא תיקמץ עם המנחה, שאם עלה בקומץ גרגר מלח או קורט לבונה הקומץ פסול.
וְאֵימָא הָכִי נָמֵי? אָמַר קְרָא: ״וְהִקְרִיבָהּ אֶל הַכֹּהֵן וְגוֹ׳ וְהִגִּישָׁהּ אֶל הַמִּזְבֵּחַ״ — מָה הַקְרָבָה אֵצֶל כֹּהֵן בִּכְלִי, אַף הַגָּשָׁה אֵצֶל מִזְבֵּחַ בִּכְלִי.
ולאחר מכן קומץ ממקום שנתרבה שמנה, שנאמר "מסלתה ומשמנה" ו"מגרשה ומשמנה", כלומר ממקום שיש בו הרבה שמן.
וּמְסַלֵּק אֶת לְבוֹנָתָהּ לְצַד אֶחָד. כִּי הֵיכִי דְּלָא תִּקְּמוֹץ בַּהֲדֵי מִנְחָה, כְּדִתְנַן: קָמַץ וְעָלָה בְּיָדוֹ צְרוֹר אוֹ גַּרְגֵּר מֶלַח אוֹ קוֹרֶט לְבוֹנָה — פָּסוּל.
ומקשים: מדוע צריך לקדש שוב בכלי שרת אם כבר קידש פעם אחת? ומשיבים: כמו דם, שהסכין מקדשת בשחיטה ובכל זאת חוזרים ומקדשים בכלי שרת בקבלת הדם, כך גם במנחה.
וְקוֹמֵץ מִמָּקוֹם שֶׁנִּתְרַבָּה שַׁמְנָהּ. מְנָלַן? דִּכְתִיב ״מִסׇּלְתָּהּ וּמִשַּׁמְנָהּ״, ״מִגִּרְשָׂהּ וּמִשַּׁמְנָהּ״.
ועוד: מלקט הלבונה ונותנה על גבי הקומץ, שנאמר "ואת כל הלבנה אשר על המנחה".
וְנוֹתְנוֹ לְתוֹךְ כְּלִי שָׁרֵת וּמְקַדְּשׁוֹ בִּכְלִי שָׁרֵת. לְמָה לִי? הָא קַדְּשַׁהּ חֲדָא זִימְנָא! מִידֵּי דְּהָוֵה אַדָּם; דָּם, אַף עַל גַּב דְּקַדֵּישְׁתֵּיהּ סַכִּין בְּצַוַּאר בְּהֵמָה, הֲדַר מְקַדֵּישׁ לֵיהּ בִּכְלִי שָׁרֵת, הָכָא נָמֵי לָא שְׁנָא.
ולגבי "מעלהו ומקטירו בכלי שרת" — ומקשים: וכי מקטירים בתוך כלי? ומשיבים: מעלהו בכלי שרת כדי להקטירו, אך ההקטרה עצמה על גבי המזבח.
וּמְלַקֵּט אֶת לְבוֹנָתָהּ וְנוֹתְנָהּ עַל גַּבָּיו — דִּכְתִיב: ״וְאֶת כׇּל הַלְּבוֹנָה אֲשֶׁר עַל הַמִּנְחָה״.
ולאחר מכן "ומולחו ונותנו על האישים", שנאמר "וכל קרבן מנחתך במלח תמלח".
וּמַעֲלֵהוּ
