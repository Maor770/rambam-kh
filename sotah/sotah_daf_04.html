וְכַמָּה שִׁיעוּר סְתִירָה? כְּדֵי טוּמְאָה, כְּדֵי בִיאָה, כְּדֵי הַעֲרָאָה.
הגמרא מגדירה כמה זמן צריכה להיות ההתייחדות כדי שתיחשב "סתירה": לפחות "כדי טומאה" = כדי ביאה = כדי העראה (מגע ראשוני בלבד, לא ביאה שלמה).
כְּדֵי הַקָּפַת דֶּקֶל, דִּבְרֵי רַבִּי יִשְׁמָעֵאל. רַבִּי אֱלִיעֶזֶר אוֹמֵר: כְּדֵי מְזִיגַת הַכּוֹס. רַבִּי יְהוֹשֻׁעַ אוֹמֵר: כְּדֵי לִשְׁתּוֹתוֹ.
מחלוקת בשיעור הזמן: ר' ישמעאל — כדי הקפת דקל (= ללכת סביב עץ דקל). ר' אליעזר — כדי מזיגת כוס. ר' יהושע — כדי לשתות כוס.
בֶּן עַזַּאי אוֹמֵר: כְּדֵי לִצְלוֹת בֵּיצָה. רַבִּי עֲקִיבָא אוֹמֵר: כְּדֵי לְגוֹמְעָהּ. רַבִּי יְהוּדָה בֶּן בְּתֵירָא אוֹמֵר: כְּדֵי לִגְמוֹעַ שָׁלֹשׁ בֵּיצִים זוֹ אַחַר זוֹ. רַבִּי אֶלְעָזָר בֶּן יִרְמְיָה אוֹמֵר: כְּדֵי לִקְשׁוֹר גַּרְדִּי נִימָא.
עוד דעות: בן עזאי — כדי לצלות ביצה. ר' עקיבא — כדי לצלות ולגמוע ביצה. ר' יהודה בן בתירא — כדי לגמוע שלוש ביצים. ר' אלעזר בן ירמיה — כדי שאורג יקשור חוט.
חָנִין בֶּן פִּנְחָס אוֹמֵר: כְּדֵי שֶׁתּוֹשִׁיט יָדָהּ לְתוֹךְ פִּיהָ לִיטּוֹל קֵיסָם. פְּלֵימוֹ אוֹמֵר: כְּדֵי שֶׁתּוֹשִׁיט יָדָהּ לַסַּל לִיטּוֹל כִּכָּר. אַף עַל פִּי שֶׁאֵין רְאָיָה לַדָּבָר, זֵכֶר לַדָּבָר: ״כִּי בְעַד אִשָּׁה זוֹנָה עַד כִּכַּר לָחֶם״.
עוד: חנין בן פנחס — כדי שתושיט יד לפיה וּתוציא קיסם. פלימו — כדי שתושיט יד לסל ותיטול ככר לחם. הפסוק "כי בעד אשה זונה עד ככר לחם" הוא זכר (רמז) לדבר.
וְכׇל הָנֵי לְמָה לִי?
שואלים: למה צריך לומר גם "כדי טומאה", גם "כדי ביאה", גם "כדי העראה"? הרי הכל אותו שיעור!
צְרִיכִי, דְּאִי תְּנָא כְּדֵי טוּמְאָה, הֲוָה אָמֵינָא: כְּדֵי טוּמְאָתָהּ וְאַרְצוּתָהּ, קָא מַשְׁמַע לַן כְּדֵי בִיאָה.
תשובה: אם רק "כדי טומאה" — הייתי חושב שכולל גם פיתוי לפני הביאה. לכן צריך "כדי ביאה" — ביאה בלבד, בלי פיתוי.
וְאִי תְּנָא כְּדֵי בִיאָה, הֲוָה אָמֵינָא: כְּדֵי גְּמַר בִּיאָה, קָא מַשְׁמַע לַן כְּדֵי הַעֲרָאָה.
ואם רק "כדי ביאה" — הייתי חושב ביאה שלמה. לכן "כדי העראה" — מגע ראשוני בלבד, לא גמר.
וְאִי אַשְׁמְעִינַן כְּדֵי הַעֲרָאָה, הֲוָה אָמֵינָא: כְּדֵי הַעֲרָאָה וְאַרְצוּתָהּ, קָא מַשְׁמַע לַן כְּדֵי טוּמְאָה. וְכַמָּה כְּדֵי [הַעֲרָאָה] — כְּדֵי הַקָּפַת דֶּקֶל.
ואם רק "כדי העראה" — הייתי חושב כולל פיתוי. לכן "כדי טומאה" — ללא פיתוי. ושיעור "כדי העראה" = כדי הקפת דקל.
וּרְמִינְהִי ״וְנִסְתְּרָה״, וְכַמָּה שִׁיעוּר סְתִירָה לֹא שָׁמַעְנוּ. כְּשֶׁהוּא אוֹמֵר ״וְהִיא נִטְמָאָה״, הֱוֵי אוֹמֵר — כְּדֵי טוּמְאָה, כְּדֵי בִיאָה, כְּדֵי הַעֲרָאָה, כְּדֵי חֲזָרַת דֶּקֶל, דִּבְרֵי רַבִּי אֱלִיעֶזֶר.
מקשים מברייתא אחרת: שם ר' אליעזר אומר "כדי חזרת דקל" (= חזרת הדקל ברוח), ואילו במשנה ר' ישמעאל אומר "כדי הקפת דקל" (= הליכה סביבו). אם שני השיעורים זהים — סתירה!
רַבִּי יְהוֹשֻׁעַ אוֹמֵר: כְּדֵי מְזִיגַת הַכּוֹס. בֶּן עַזַּאי אוֹמֵר: כְּדֵי לִשְׁתּוֹתוֹ. רַבִּי עֲקִיבָא אוֹמֵר: כְּדֵי לִצְלוֹת בֵּיצָה. רַבִּי יְהוּדָה בֶּן בְּתֵירָא אוֹמֵר: כְּדֵי לְגוֹמְעָהּ.
גם בברייתא ר' יהושע אומר "כדי מזיגת הכוס" (לא "כדי לשתותו" כבמשנה), ובן עזאי אומר "כדי לשתותו" (לא "כדי לצלות ביצה" כבמשנה).
קָא סָלְקָא דַּעְתִּין: הַיְינוּ הַקָּפַת דֶּקֶל, הַיְינוּ חֲזָרַת דֶּקֶל. הָתָם אָמַר רַבִּי יִשְׁמָעֵאל: כְּדֵי הַקָּפַת דֶּקֶל, וּפְלִיג רַבִּי אֱלִיעֶזֶר עֲלֵיהּ. הָכָא אָמַר רַבִּי אֱלִיעֶזֶר כְּדֵי חֲזָרַת דֶּקֶל!
אביי מתרץ: "הקפת דקל" = הליכה סביב דקל ברגל. "חזרת דקל" = חזרת הדקל למקומו אחרי שנדחה ברוח. שני שיעורים שונים.
אָמַר אַבָּיֵי: הַקָּפָה בָּרֶגֶל, חֲזָרָה בָּרוּחַ.
רב אשי שואל: "חזרת דקל" ברוח — עד שהדקל הולך וחוזר? או עד שנח לגמרי במקומו? תיקו.
בָּעֵי רַב אָשֵׁי: חֲזָרָה בָּרוּחַ כִּי הֵיכִי דְּאָזֵיל וַהֲדַר אָתֵי, אוֹ דִילְמָא כִּי הֵיכִי דְּאָזֵיל וְאָתֵי וַהֲדַר קָאֵי בְּדוּכְתֵּיהּ. תֵּיקוּ.
עוד סתירה: ר' אליעזר במשנה אומר "כדי מזיגת הכוס", בברייתא "כדי חזרת דקל". תשובה: שני הביטויים הם אותו שיעור זמן.
הָתָם אָמַר רַבִּי אֱלִיעֶזֶר: כְּדֵי מְזִיגַת הַכּוֹס, הָכָא: כְּדֵי חֲזָרַת דֶּקֶל! אִידֵּי וְאִידֵּי חַד שִׁיעוּרָא הוּא.
ר' יהושע: במשנה "כדי לשתותו", בברייתא "כדי מזיגת הכוס". פתרון: כוונתו למזוג ולשתות ביחד. ולמה לא שיעור אחד? כי אז ר' יהושע = ר' אליעזר.
הָתָם אָמַר רַבִּי יְהוֹשֻׁעַ: כְּדֵי לִשְׁתּוֹתוֹ, הָכָא אָמַר: כְּדֵי מְזִיגַת הַכּוֹס! אֵימָא כְּדֵי לִמְזוֹג וְלִשְׁתּוֹת. וְלֵימָא: אִידֵּי וְאִידֵּי חַד שִׁיעוּרָא הוּא! אִם כֵּן, הַיְינוּ רַבִּי אֱלִיעֶזֶר.
בן עזאי: במשנה "כדי לצלות ביצה", בברייתא "כדי לשתותו". פתרון: שניהם אותו שיעור — צליית ביצה = שתיית כוס.
הָתָם אָמַר בֶּן עַזַּאי: כְּדֵי לִצְלוֹת בֵּיצָה, הָכָא אָמַר: כְּדֵי לִשְׁתּוֹתוֹ. אִידֵּי וְאִידֵּי חַד שִׁיעוּרָא הוּא.
ר' עקיבא: במשנה "כדי לגומעה", בברייתא "כדי לצלות ביצה". פתרון: כוונתו לצלות ולגמוע. ולמה לא שיעור אחד? כי אז ר' עקיבא = בן עזאי.
הָתָם אָמַר רַבִּי עֲקִיבָא: כְּדֵי לְגוֹמְעָהּ, הָכָא אָמַר: כְּדֵי לִצְלוֹת בֵּיצָה! אֵימָא כְּדֵי לִצְלוֹת בֵּיצָה וּלְגוֹמְעָהּ. וְלֵימָא: אִידֵּי וְאִידֵּי חַד שִׁיעוּרָא הוּא! אִם כֵּן, הַיְינוּ בֶּן עַזַּאי.
ר' יהודה בן בתירא: במשנה "כדי לגמוע שלוש ביצים", בברייתא "כדי לגומעה" אחת. פתרון: בברייתא מגיב לר' עקיבא — גמיעת שלוש ביצים = צלייה + גמיעה.
הָתָם אָמַר רַבִּי יְהוּדָה בֶּן בְּתֵירָא: כְּדֵי לִגְמוֹעַ שָׁלֹשׁ בֵּיצִים זוֹ אַחַר זוֹ, הָכָא אָמַר: כְּדֵי לְגוֹמְעָהּ! לִדְבָרָיו דְּרַבִּי עֲקִיבָא קָאָמַר, דְּקָאָמַר מְשַׁעֲרִין בִּצְלִיאָה וּבִגְמִיעָה: אֵימָא שִׁיעוּר גְּמִיעָה לְחוֹדַהּ כְּדֵי לִגְמֹעַ שָׁלֹשׁ בֵּיצִים זוֹ אַחַר זוֹ, דְּהַיְינוּ צְלִיאָה וּגְמִיעָה.
ר' אלעזר בן ירמיה: כדי שאורג יקשור חוט. רב אשי שואל: חוט רחוק (לוקח יותר זמן) או קרוב? תיקו.
רַבִּי אֶלְעָזָר בֶּן יִרְמְיָה אוֹמֵר: כְּדֵי שֶׁיִּקְשׁוֹר גַּרְדִּי נִימָא. בָּעֵי רַב אָשֵׁי: דִּמְרַחַק, אוֹ דִּמְקָרַב? תֵּיקוּ.
חנין בן פנחס: כדי שתוציא קיסם מפיה. רב אשי שואל: קיסם תקוע חזק או לא? תיקו.
חָנִין בֶּן פִּנְחָס אוֹמֵר: כְּדֵי שֶׁתּוֹשִׁיט יָדָהּ לְתוֹךְ פִּיהָ לִיטּוֹל קֵיסָם. בָּעֵי רַב אָשֵׁי: דִּמְהַדַּק אוֹ דְּלָא מְהַדַּק? תֵּיקוּ.
פלימו: כדי שתיטול ככר לחם מסל. רב אשי שואל שורת שאלות: הככר מהודק בסל או רופף? סל חדש או ישן? לחם חם או קר?
פְּלֵימוֹ אוֹמֵר: כְּדֵי שֶׁתּוֹשִׁיט יָדָהּ לַסַּל לִיטּוֹל כִּכָּר. בָּעֵי רַב אָשֵׁי: דִּמְהַדַּק אוֹ דְּלָא מְהַדַּק? בְּחַדְתָּא אוֹ בְּעַתִּיקָא? בְּחַמִּימָא אוֹ בְּקָרִירָא?
ועוד שאלות: ככר חיטים או שעורים? רכה או קשה? כל השאלות נשארו ללא תשובה — תיקו.
בִּדְחִטֵּי אוֹ בְּדִשְׂעָרֵי? בְּרַכִּיכָא אוֹ בַּאֲקוֹשָׁא? תֵּיקוּ.
המשך שאלות רב אשי: ככר חיטים או שעורים? רכה או קשה? תיקו.
אָמַר רַב יִצְחָק בַּר רַב יוֹסֵף אָמַר רַבִּי יוֹחָנָן: כׇּל אֶחָד וְאֶחָד — בְּעַצְמוֹ שִׁיעֵר. וְהָאִיכָּא בֶּן עַזַּאי דְּלָא נְסֵיב?
ר' יוחנן: כל חכם שיער את זמן ההעראה מניסיונו האישי. קושיה: בן עזאי לא נשא אשה — איך שיער מעצמו?
אִיבָּעֵית אֵימָא: נְסֵיב וּפֵירַשׁ הֲוָה. וְאִיבָּעֵית אֵימָא: מֵרַבֵּיהּ שְׁמִיעַ לֵיהּ. וְאִיבָּעֵית אֵימָא: ״סוֹד ה׳ לִירֵאָיו״.
שלוש תשובות: ① נשא אשה ופירש ממנה. ② שמע מרבו. ③ "סוד ה' ליראיו" — ידע ברוח הקודש דברים שלא התנסה בהם.
דָּרֵשׁ רַב עַוִּירָא, זִמְנִין אָמַר לַהּ מִשְּׁמֵיהּ דְּרַבִּי אַמֵּי, וְזִמְנִין אָמַר לַהּ מִשְּׁמֵיהּ דְּרַבִּי אַסִּי: כׇּל הָאוֹכֵל לֶחֶם בְּלֹא נְטִילַת יָדַיִם — כְּאִילּוּ בָּא עַל אִשָּׁה זוֹנָה. שֶׁנֶּאֱמַר: ״כִּי בְעַד אִשָּׁה זוֹנָה עַד כִּכַּר לָחֶם״.
דרשה מהפסוק "כי בעד אשה זונה עד ככר לחם": האוכל לחם בלי נטילת ידיים — כאילו בא על זונה. הקשר: "ככר לחם" = אכילה בלי נקיות.
אָמַר רָבָא: הַאי ״בְּעַד אִשָּׁה זוֹנָה עַד כִּכַּר לָחֶם״ — ״בְּעַד כִּכַּר לֶחֶם עַד אִשָּׁה זוֹנָה״ מִיבְּעֵי לֵיהּ! אֶלָּא אָמַר רָבָא: כׇּל הַבָּא עַל אִשָּׁה זוֹנָה, לְסוֹף מְבַקֵּשׁ כִּכַּר לֶחֶם.
רבא דוחה: הפסוק מדבר הפוך! לא שככר גורמת לזנות, אלא שזנות גורמת לעוני. כלומר: הבא על זונה — בסוף מחפש ככר לחם.
אָמַר רַבִּי זְרִיקָא אָמַר רַבִּי אֶלְעָזָר: כׇּל הַמְּזַלְזֵל בִּנְטִילַת יָדַיִם נֶעֱקָר מִן הָעוֹלָם. אָמַר רַב חִיָּיא בַּר אָשֵׁי אָמַר רַב: מַיִם רִאשׁוֹנִים — צָרִיךְ שֶׁיַּגְבִּיהַּ יָדָיו לְמַעְלָה, מַיִם אַחֲרוֹנִים — צָרִיךְ שֶׁיַּשְׁפִּיל יָדָיו לְמַטָּה. תַּנְיָא נָמֵי הָכִי: הַנּוֹטֵל יָדָיו — צָרִיךְ שֶׁיַּגְבִּיהַּ יָדָיו לְמַעְלָה, שֶׁמָּא יֵצְאוּ הַמַּיִם חוּץ לַפֶּרֶק, וְיַחְזְרוּ וִיטַמְּאוּ אֶת הַיָּדַיִם.
ר' אלעזר: המזלזל בנטילת ידיים — נעקר מהעולם. רב: מים ראשונים (לפני אכילה) — מגביה ידיו שלא יחזרו המים ויטמאו. מים אחרונים — משפיל למטה.
אָמַר רַבִּי אֲבָהוּ: כׇּל הָאוֹכֵל פַּת בְּלֹא נִיגּוּב יָדַיִם — כְּאִילּוּ אוֹכֵל לֶחֶם טָמֵא, שֶׁנֶּאֱמַר: ״וַיֹּאמֶר ה׳ כָּכָה יֹאכְלוּ בְנֵי יִשְׂרָאֵל אֶת לַחְמָם טָמֵא וְגוֹ׳״.
ר' אבהו: האוכל פת בלי ניגוב ידיים — כאילו אוכל לחם טמא. דרשה מהפסוק "ככה יאכלו בני ישראל את לחמם טמא".
וּמַאי ״וְאֵשֶׁת אִישׁ נֶפֶשׁ יְקָרָה תָצוּד״? אָמַר רַבִּי חִיָּיא בַּר אַבָּא אָמַר רַבִּי יוֹחָנָן: כׇּל אָדָם שֶׁיֵּשׁ בּוֹ גַּסּוּת הָרוּחַ — לְבַסּוֹף נִכְשָׁל בְּאֵשֶׁת אִישׁ, שֶׁנֶּאֱמַר: ״וְאֵשֶׁת אִישׁ נֶפֶשׁ יְקָרָה תָצוּד״.
המשך הפסוק "ואשת איש נפש יקרה תצוד" — ר' יוחנן: מי שיש בו גסות הרוח נכשל באשת איש. "נפש יקרה" = נפש מתנשאת.
אָמַר רָבָא: הַאי ״נֶפֶשׁ יְקָרָה״ — ״נֶפֶשׁ גְּבוֹהָה״ מִיבְּעֵי לֵיהּ. וְעוֹד: ״הִיא תָּצוּד״ מִיבְּעֵי לֵיהּ! אֶלָּא אָמַר רָבָא: כׇּל הַבָּא עַל אֵשֶׁת אִישׁ, אֲפִילּוּ לָמַד תּוֹרָה דִּכְתִיב בַּהּ ״יְקָרָה הִיא מִפְּנִינִים״, מִכֹּהֵן גָּדוֹל שֶׁנִּכְנָס לִפְנַי וְלִפְנִים, ״הִיא תְּצוּדֶנּוּ״ — לְדִינָהּ שֶׁל גֵּיהִנָּם.
רבא דוחה: "יקרה" לא מתאים לגאווה, צריך "גבוהה"! אלא: הבא על אשת איש — אפילו למד תורה ("יקרה מפנינים") ונחשב ככהן גדול — "היא תצודנו" לדינה של גיהנם.
אָמַר רַבִּי יוֹחָנָן מִשּׁוּם רַבִּי שִׁמְעוֹן בֶּן יוֹחַי: כׇּל אָדָם שֶׁיֵּשׁ בּוֹ גַּסּוּת הָרוּחַ — כְּאִילּוּ עוֹבֵד עֲבוֹדָה זָרָה. כְּתִיב הָכָא: ״תּוֹעֲבַת ה׳ כׇּל גְּבַהּ לֵב״, וּכְתִיב הָתָם: ״וְלָא תָבִיא תוֹעֵבָה אֶל בֵּיתֶךָ״.
ר' יוחנן בשם רשב"י: גסות הרוח = כאילו עובד עבודה זרה. גזירה שווה: "תועבת ה'" נאמר גם על גאווה וגם על עבודה זרה.
וְרַבִּי יוֹחָנָן דִּידֵיהּ אָמַר: כְּאִילּוּ כָּפַר בָּעִיקָּר, שֶׁנֶּאֱמַר: ״וְרָם לְבָבֶךָ וְשָׁכַחְתָּ אֶת ה׳ אֱלֹהֶיךָ וְגוֹ׳״.
ר' יוחנן עצמו: גסות הרוח = כאילו כפר בה'. "ורם לבבך ושכחת את ה' אלוהיך" — התנשאות מובילה לשכחת ה'.
רַבִּי חָמָא בַּר חֲנִינָא אָמַר: כְּאִילּוּ בָּא עַל כׇּל הָעֲרָיוֹת. כְּתִיב הָכָא: ״תּוֹעֲבַת ה׳ כׇּל גְּבַהּ לֵב״, וּכְתִיב הָתָם: ״כִּי אֶת כׇּל הַתּוֹעֵבוֹת הָאֵל וְגוֹ׳״.
ר' חמא בר חנינא: גסות הרוח = כאילו בא על כל העריות. גזירה שווה: "תועבת ה'" (גאווה) = "כי את כל התועבות האל" (עריות).
עוּלָּא אָמַר: כְּאִילּוּ בָּנָה בָּמָה, שֶׁנֶּאֱמַר: ״חִדְלוּ לָכֶם מִן הָאָדָם אֲשֶׁר נְשָׁמָה בְּאַפּוֹ כִּי בַּמֶּה נֶחְשָׁב הוּא״ — אַל תִּיקְרֵי ״בַּמֶּה״, אֶלָּא ״בָּמָה״.
עולא: גסות הרוח = כאילו בנה במה. דרשה: "כי במה נחשב הוא" — אל תקרי "בַּמֶּה" אלא "בָּמָה" (= במת עבודה זרה).
מַאי ״יָד לְיָד לֹא יִנָּקֶה״? אָמַר רַב: כׇּל הַבָּא עַל אֵשֶׁת אִישׁ, אֲפִילּוּ הִקְנָהוּ לְהַקָּדוֹשׁ בָּרוּךְ הוּא שָׁמַיִם וָאָרֶץ כְּאַבְרָהָם אָבִינוּ, דִּכְתִיב בֵּיהּ: ״הֲרִימֹתִי יָדִי אֶל ה׳ אֵל עֶלְיוֹן קֹנֵה שָׁמַיִם וָאָרֶץ״ — לֹא יִנָּקֶה מִדִּינָהּ שֶׁל גֵּיהִנָּם.
"יד ליד לא יינקה" (משלי יא) — רב: הבא על אשת איש, אפילו הקנה לה' שמים וארץ כאברהם ("הרימותי ידי") — לא יינקה מגיהנם.
קַשְׁיָא לְהוּ לִדְבֵי רַבִּי שֵׁילָא: הַאי ״יָד לְיָד לֹא יִנָּקֶה״, ״יָדִי״ מִיבְּעֵי לֵיהּ!
קושיה מבית ר' שילא: אם מדובר באברהם, צריך "ידי" ולא "יד ליד"!
אֶלָּא אָמְרִי דְּבֵי רַבִּי שֵׁילָא: אֲפִילּוּ קִיבֵּל תּוֹרָה כְּמֹשֶׁה רַבֵּינוּ, דִּכְתִיב בֵּיהּ ״מִימִינוֹ אֵשׁ דָּת לָמוֹ״ — לֹא יִנָּקֶה מִדִּינָהּ שֶׁל גֵּיהִנָּם.
דרשה חלופית (בית ר' שילא): אפילו קיבל תורה כמשה ("מימינו אש דת למו" = יד ליד) — לא יינקה מגיהנם.
קַשְׁיָא לֵיהּ לְרַבִּי יוֹחָנָן: הַאי ״יָד לְיָד״, ״יָד מִיָּד״ מִיבַּעְיָא לֵיהּ!
ר' יוחנן מקשה: "יד ליד" = מיד ליד? צריך "יד מיד" (= קבלה מיד)!
אֶלָּא אָמַר רַבִּי יוֹחָנָן:
ר' יוחנן דורש: אפילו עושה צדקה בסתר (יד ליד = מידו ליד העני) — מי שיש בו גסות הרוח לא יינקה מגיהנם.
