טקסט מקורי + ביאור מתומצת מבוסס על ביאור שטיינזלץ
31 קטעים
שְׁלִישִׁי הַבָּא מֵחֲמַת שֵׁנִי, דְּשֵׁנִי גּוּפֵיהּ אָסוּר בְּחוּלִּין — אֵינוֹ דִּין שֶׁעוֹשֶׂה רְבִיעִי בַּקּוֹדֶשׁ?
המשך הקושיה: אם ר' יוסי סובר כאבא שאול, היה לו ללמוד רביעי בקודש מאוכל שבא מטבול יום (שלאבא שאול עושה רביעי). ואז הקל וחומר חזק יותר ולא ניתן לפריכה.
וְכִי תֵּימָא מִשּׁוּם דְּאִיכָּא לְמִיפְרַךְ: מָה לִטְבוּל יוֹם שֶׁכֵּן אַב הַטּוּמְאָה — הָא אַיְיתִינֵהּ מִמְּחוּסַּר כִּיפּוּרִים, וְלָא פַּרְכֵיהּ!
פירכה אפשרית ("שכן אב הטומאה") — אינה רלוונטית, שהרי ר' יוסי עצמו הביא קל וחומר ממחוסר כיפורים (שגם הוא היה אב) ולא חשש לפירכה זו.
אָמַר רַב אַסִּי אָמַר רַב, וְאָמְרִי לַהּ אָמַר רַבָּה בֶּן אִיסִי אָמַר רַב: רַבִּי מֵאִיר, וְרַבִּי יוֹסֵי, וְרַבִּי יְהוֹשֻׁעַ, וְרַבִּי אֶלְעָזָר, וְרַבִּי אֱלִיעֶזֶר — כּוּלְּהוּ סְבִירָא לְהוּ דְּאֵין שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחוּלִּין.
רב אסי אמר רב: חמישה תנאים סוברים שאין שני לטומאה עושה שלישי בחולין — ר' מאיר, ר' יוסי, ר' יהושע, ר' אלעזר, ור' אליעזר.
רַבִּי מֵאִיר — דִּתְנַן: כׇּל הַטָּעוּן בִּיאַת מַיִם מִדִּבְרֵי סוֹפְרִים, מְטַמֵּא אֶת הַקּוֹדֶשׁ, וּפוֹסֵל אֶת הַתְּרוּמָה, וּמוּתָּר בַּחוּלִּין וּבַמַּעֲשֵׂר, דִּבְרֵי רַבִּי מֵאִיר. וַחֲכָמִים אוֹסְרִין בַּמַּעֲשֵׂר.
ר' מאיר — ממשנה: "כל הטעון ביאת מים מדברי סופרים — מטמא קודש, פוסל תרומה, ומותר בחולין ובמעשר". מותר בחולין = אין שני עושה שלישי בחולין.
רַבִּי יוֹסֵי — הָא דַּאֲמַרַן. דְּאִם אִיתֵיהּ — לַיְיתֵיהּ לִרְבִיעִי בַּתְּרוּמָה, וַחֲמִישִׁי בַּקּוֹדֶשׁ.
ר' יוסי — כבר הוכחנו: אם היה שני עושה שלישי בחולין, היה יכול ללמוד גם רביעי בתרומה וחמישי בקודש (מקל וחומר). מזה שלא למד — ברור שאינו סובר כך.
רַבִּי יְהוֹשֻׁעַ — דִּתְנַן, רַבִּי אֱלִיעֶזֶר אוֹמֵר: הָאוֹכֵל אוֹכֶל רִאשׁוֹן רִאשׁוֹן, שֵׁנִי — שֵׁנִי, שְׁלִישִׁי — שְׁלִישִׁי. רַבִּי יְהוֹשֻׁעַ אוֹמֵר: הָאוֹכֵל אוֹכֶל רִאשׁוֹן וְאוֹכֶל שֵׁנִי — שֵׁנִי, שְׁלִישִׁי — שֵׁנִי בַּקּוֹדֶשׁ, וְאֵין שֵׁנִי בַּתְּרוּמָה.
ר' יהושע — ממשנה: "האוכל אוכל ראשון ושני — נעשה שני. שלישי — נעשה שני בקודש ואין שני בתרומה". אבל דין זה נאמר רק בחולין שנעשו על טהרת תרומה.
בְּחוּלִּין שֶׁנַּעֲשׂוּ עַל טׇהֳרַת תְּרוּמָה.
דיוק: "על טהרת תרומה" — כן, אבל "על טהרת הקודש" — לא. חולין רגילים = דינם כחולין, ואין בהם שלישי.
עַל טׇהֳרַת הַתְּרוּמָה — אִין, עַל טׇהֳרַת הַקּוֹדֶשׁ — לָא.
מכאן שר' יהושע סובר: אין שני עושה שלישי בחולין רגילים.
אַלְמָא קָסָבַר: אֵין שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחוּלִּין.
ר' אלעזר — מברייתא: "שלושתם שווים" — ראשון בקודש, בחולין ובתרומה כולם מטמאים. ההבדל: בקודש — מטמא שניים ופוסל אחד (= רביעי). בתרומה — מטמא אחד ופוסל אחד (= שלישי). בחולין — פוסל אחד בלבד (= שני). → אין שלישי בחולין.
רַבִּי אֶלְעָזָר — דְּתַנְיָא, רַבִּי אֶלְעָזָר אוֹמֵר: שְׁלָשְׁתָּן שָׁוִין, הָרִאשׁוֹן שֶׁבַּקּוֹדֶשׁ וְשֶׁבַּחוּלִּין וְשֶׁבַּתְּרוּמָה —
ר' אלעזר — בברייתא: ראשון לטומאה בקודש — מטמא שניים ופוסל אחד. בתרומה — מטמא אחד ופוסל אחד.
מְטַמֵּא שְׁנַיִם, וּפוֹסֵל אֶחָד בַּקּוֹדֶשׁ,
בחולין — פוסל אחד בלבד (= שני). → אין שלישי בחולין.
מְטַמֵּא אֶחָד, וּפוֹסֵל אֶחָד בַּתְּרוּמָה,
ר' אליעזר — ממשנה: חלה ניטלת מעיסה טהורה על עיסה טמאה. כיצד? שתי עיסות, אחת טהורה ואחת טמאה — נותן פחות מכביצה באמצע (כדי ליטול מן המוקף).
וּפוֹסֵל אֶחָד בַּחוּלִּין.
בברייתא — אפילו כביצה מותר. "כביצה" = שיעור שמטמא. אם מניח אוכל בשיעור כביצה ליד עיסה טמאה (ראשונה) — נעשה שני. אם שני היה עושה שלישי בחולין — היה מטמא את העיסה הטהורה!
רַבִּי אֱלִיעֶזֶר, דִּתְנַן, רַבִּי אֱלִיעֶזֶר אוֹמֵר: חַלָּה נִיטֶּלֶת מִן הַטְּהוֹרָה עַל הַטְּמֵאָה, כֵּיצַד? שְׁתֵּי עִיסּוֹת אַחַת טְהוֹרָה וְאַחַת טְמֵאָה, נוֹטֵל כְּדֵי חַלָּה מֵעִיסָּה שֶׁלֹּא הוּרְמָה חַלָּתָהּ, וְנוֹתֵן פָּחוֹת מִכְּבֵיצָה בָּאֶמְצַע כְּדֵי לִיטּוֹל מִן הַמּוּקָּף.
מזה שר' אליעזר מתיר — מוכח שסובר: אין שני עושה שלישי בחולין.
וַחֲכָמִים אוֹסְרִין.
וחכמים אוסרים לעשות כן (= לחבר עיסה טהורה וטמאה).
וְתַנְיָא כְּבֵיצָה.
ובברייתא: ר' אליעזר מתיר אפילו כביצה (= שיעור המטמא אוכלים).
סַבְרוּהָ אִידֵּי וְאִידֵּי בְּעִיסָּה רִאשׁוֹנָה, וְחוּלִּין הַטְּבוּלִין לְחַלָּה לָא כְּחַלָּה דָּמוּ.
הנחות: ① שתי הגרסאות (משנה וברייתא) מדברות בעיסה ראשונה לטומאה. ② חולין הטבולים לחלה = כחולין רגילים (לא כתרומה).
מַאי לָאו בְּהָא קָמִיפַּלְגִי, דְּמָר סָבַר: אֵין שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחוּלִּין,
לפי הנחות אלו: ר' אליעזר סובר "אין שני עושה שלישי בחולין" — לכן מותר להניח כביצה. חכמים סוברים "שני עושה שלישי" — ולכן אוסרים.
וּמָר סָבַר: שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחוּלִּין.
רב מרי חולק: הכל מסכימים שאין שני עושה שלישי בחולין. המחלוקת היא אחרת — האם "חולין הטבולים לחלה כחלה דמו" (= דינם כתרומה). חכמים = כחלה (ולכן אוסרים). ר' אליעזר = לא כחלה.
אָמַר רַב מָרִי בְּרֵיהּ דְּרַב כָּהֲנָא: דְּכוּלֵּי עָלְמָא אֵין שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחוּלִּין, וְהָכָא בְּחוּלִּין הַטְּבוּלִין לְחַלָּה קָמִיפַּלְגִי. מָר סָבַר: כְּחַלָּה דָּמוּ, וּמָר סָבַר: לָא כְּחַלָּה דָּמוּ.
הסבר שלישי: הכל מסכימים שלא כחלה דמו, ואין שני עושה שלישי. המחלוקת — האם מותר לגרום טומאה לחולין בארץ ישראל.
וְאִיבָּעֵית אֵימָא: דְּכוּלֵּי עָלְמָא חוּלִּין הַטְּבוּלִין לְחַלָּה לָא כְּחַלָּה דָּמוּ, וְאֵין שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחוּלִּין, וְהָכָא בְּמוּתָּר לִגְרוֹם טוּמְאָה לְחוּלִּין שֶׁבְּאֶרֶץ יִשְׂרָאֵל קָמִיפַּלְגִי.
ר' אליעזר — מותר לגרום טומאה לחולין שבא"י, ולכן מותר להניח כביצה אף שתיטמא.
מָר סָבַר: מוּתָּר לִגְרוֹם טוּמְאָה לְחוּלִּין שֶׁבְּאֶרֶץ יִשְׂרָאֵל, וּמָר סָבַר: אָסוּר לִגְרוֹם טוּמְאָה לְחוּלִּין שֶׁבְּאֶרֶץ יִשְׂרָאֵל.
חכמים — אסור לגרום טומאה לחולין שבא"י, ולכן אוסרים.
בּוֹ בַּיּוֹם דָּרַשׁ רַבִּי עֲקִיבָא וְכוּ׳.
שלוש דרכים להסביר מחלוקת אחת: ① שני בחולין, ② חולין הטבולים לחלה, ③ גרימת טומאה לחולין בא"י.
בְּמַאי קָא מִיפַּלְגִי? מָר סָבַר: תְּחוּמִין דְּאוֹרָיְיתָא, וּמָר סָבַר: דְּרַבָּנַן.
משנה: באותו יום דרש ר' עקיבא לגבי תחום שבת מדין ערי הלויים, ור' אליעזר בנו של ר' יוסי הגלילי דרש את הפסוק לעניין אחר.
תָּנוּ רַבָּנַן, בּוֹ בַּיּוֹם דָּרַשׁ רַבִּי עֲקִיבָא: בְּשָׁעָה שֶׁעָלוּ יִשְׂרָאֵל מִן הַיָּם, נָתְנוּ עֵינֵיהֶם לוֹמַר שִׁירָה. וְכֵיצַד אָמְרוּ שִׁירָה — כְּגָדוֹל הַמַּקְרֶא אֶת הַלֵּל, וְהֵן עוֹנִין אַחֲרָיו רָאשֵׁי פְרָקִים. מֹשֶׁה אָמַר ״אָשִׁירָה לַה׳״, וְהֵן אוֹמְרִים ״אָשִׁירָה לַה׳״, מֹשֶׁה אָמַר ״כִּי גָאֹה גָּאָה״ וְהֵן אוֹמְרִים ״אָשִׁירָה לַה׳״.
במה חלוקים? ר' עקיבא — תחומי שבת מדאורייתא (ולכן דורש את הפסוק לתחומין). ר' אליעזר בר' יוסי הגלילי — תחומין מדרבנן (ולכן דורש את הפסוק לעניין אחר).
רַבִּי אֱלִיעֶזֶר בְּנוֹ שֶׁל רַבִּי יוֹסֵי הַגְּלִילִי אוֹמֵר: כְּקָטָן הַמַּקְרֶא אֶת הַלֵּל, וְהֵן עוֹנִין אַחֲרָיו כׇּל מָה שֶׁהוּא אוֹמֵר. מֹשֶׁה אָמַר ״אָשִׁירָה לַה׳״ וְהֵן אוֹמְרִים ״אָשִׁירָה לַה׳״, מֹשֶׁה אָמַר ״כִּי גָאֹה גָּאָה״, וְהֵן אוֹמְרִים ״כִּי גָאֹה גָּאָה״.
ר' עקיבא: כגדול המקרא את ההלל — הוא אומר כל פסוק, והקהל עונה רק את הפזמון. כך: משה אמר "אשירה לה'", ישראל ענו "אשירה לה'". משה אמר "כי גאה גאה" — וישראל חזרו על "אשירה לה'".
רַבִּי נְחֶמְיָה אוֹמֵר: כְּסוֹפֵר הַפּוֹרֵס עַל שְׁמַע בְּבֵית הַכְּנֶסֶת, שֶׁהוּא פּוֹתֵחַ תְּחִילָּה וְהֵן עוֹנִין אַחֲרָיו.
ר' אליעזר בנו של ר' יוסי הגלילי: כקטן המקרא את ההלל — הקהל חוזר על כל מילה. משה אמר "אשירה לה'" — וישראל אמרו "אשירה לה'". משה אמר "כי גאה גאה" — וישראל אמרו "כי גאה גאה".
בְּמַאי קָמִיפַּלְגִי — רַבִּי עֲקִיבָא סָבַר: ״לֵאמֹר״, אַמִּילְּתָא קַמַּיְיתָא.
ר' נחמיה: כסופר (= חזן) הפורס על שמע — הוא פותח תחילה, והקהל ממשיכים ומשלימים בעצמם.
וְרַבִּי אֱלִיעֶזֶר בְּנוֹ שֶׁל רַבִּי יוֹסֵי הַגְּלִילִי סָבַר: ״לֵאמֹר״ — אַכֹּל מִילְּתָא וּמִילְּתָא, וְרַבִּי נְחֶמְיָה סָבַר: ״וַיֹּאמְרוּ״ — דַּאֲמוּר כּוּלְּהוּ בַּהֲדֵי הֲדָדֵי, ״לֵאמֹר״ — דִּפְתַח מֹשֶׁה בְּרֵישָׁא.
במה חלוקים? ר' עקיבא: "לאמר" = רק על המילים הראשונות ("אשירה לה'"). ר' אליעזר: "לאמר" = על כל פסוק ופסוק. ר' נחמיה: "ויאמרו" = כולם אמרו יחד, "לאמר" = שמשה פתח תחילה.
תָּנוּ רַבָּנַן, דָּרַשׁ רַבִּי יוֹסֵי הַגְּלִילִי: בְּשָׁעָה שֶׁעָלוּ יִשְׂרָאֵל מִן הַיָּם נָתְנוּ עֵינֵיהֶם לוֹמַר שִׁירָה. וְכֵיצַד אָמְרוּ שִׁירָה — עוֹלָל מוּטָּל עַל בִּרְכֵּי אִמּוֹ, וְתִינוֹק יוֹנֵק מִשְּׁדֵי אִמּוֹ. כֵּיוָן שֶׁרָאוּ אֶת הַשְּׁכִינָה, עוֹלָל הִגְבִּיהַּ צַוָּארוֹ, וְתִינוֹק שָׁמַט דַּד מִפִּיו, וְאָמְרוּ ״זֶה אֵלִי וְאַנְוֵהוּ״, שֶׁנֶּאֱמַר: ״מִפִּי עוֹלְלִים וְיֹנְקִים יִסַּדְתָּ עֹז״.
ר' יוסי הגלילי: כשעלו מהים — עולל על ברכי אמו ותינוק יונק משדי אמו. כשראו את השכינה — העולל הרים צווארו והתינוק שמט את הדד, ואמרו "זה אלי ואנוהו". מקור: "מפי עוללים ויונקים יסדת עוז".

ר' מאיר: אפילו עוברים שבמעי אמם אמרו שירה — שנאמר "במקהלות ברכו אלוהים ה' ממקור ישראל" — "ממקור" = מהרחם.
