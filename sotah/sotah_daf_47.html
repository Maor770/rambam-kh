43 קטעים
מה בדף הזה?
אגדות: נס הדובים, שלושה חולאים של אלישע, מעשה גיחזי, ומעשה יהושע בן פרחיה. משניות: דין עגלה כשנמצא הרוצח, ביטול עגלה ערופה ומי סוטה, ותקנות יוחנן כהן גדול. אגדה: רשימת ירידת הדורות — "משרבו...".
רַב וּשְׁמוּאֵל, חַד אָמַר: נֵס. וְחַד אָמַר: נֵס בְּתוֹךְ נֵס. מַאן דְּאָמַר נֵס — יַעַר הֲוָה, דּוּבִּים לָא הֲווֹ. מַאן דְּאָמַר נֵס בְּתוֹךְ נֵס — לֹא יַעַר הֲוָה וְלָא דּוּבִּים הֲווֹ. וְלִיהְוֵי דּוּבִּים וְלָא לֶיהֱוֵי יַעַר! דִּבְעִיתִי.
המשך: מחלוקת רב ושמואל על הדובים: ① נס = יער היה, דובים נבראו. ② נס בתוך נס = גם היער וגם הדובים נבראו. מדוע צריך יער? כי דובים מפחדים בשטח גלוי ולא היו תוקפים.
אָמַר רַבִּי חֲנִינָא: בִּשְׁבִיל אַרְבָּעִים וּשְׁנַיִם קׇרְבָּנוֹת שֶׁהִקְרִיב בָּלָק מֶלֶךְ מוֹאָב הוּבְקְעוּ מִיִּשְׂרָאֵל אַרְבָּעִים וּשְׁנַיִם יְלָדִים. אִינִי?
ר' חנינא: בזכות 42 קרבנות שהקריב בלק — הובקעו מישראל 42 ילדים. קושיה: הרי מקרבנות בלק יצאה גם רות!
וְהָאָמַר רַב יְהוּדָה אָמַר רַב: לְעוֹלָם יַעֲסוֹק אָדָם בְּתוֹרָה וּבְמִצְוֹת, וְאַף עַל פִּי שֶׁלֹּא לִשְׁמָהּ. שֶׁמִּתּוֹךְ שֶׁלֹּא לִשְׁמָהּ — בָּא לִשְׁמָהּ. שֶׁבִּשְׂכַר אַרְבָּעִים וּשְׁנַיִם קׇרְבָּנוֹת שֶׁהִקְרִיב בָּלָק מֶלֶךְ מוֹאָב, זָכָה וְיָצְתָה מִמֶּנּוּ רוּת שֶׁיָּצָא מִמֶּנָּה שְׁלֹמֹה, שֶׁכָּתוּב בֵּיהּ ״אֶלֶף עֹלוֹת יַעֲלֶה שְׁלֹמֹה״. וְאָמַר רַבִּי יוֹסֵי בֶּן חוֹנִי: רוּת בִּתּוֹ שֶׁל עֶגְלוֹן בְּנוֹ שֶׁל בָּלָק הָיְתָה! תַּאֲוָתוֹ מִיהָא לִקְלָלָה הֱוֵי.
תשובה: רב יהודה אמר רב — יעסוק אדם בתורה אפילו שלא לשמה, כי מתוך שלא לשמה בא לשמה. בלק הקריב ויצאה ממנו רות → שלמה שהקריב "אלף עולות". ר' יוסי: רות בתו של עגלון בנו של בלק. אבל כוונת בלק הייתה לקללה — ולכן גרם גם נזק.
״וַיֹּאמְרוּ אַנְשֵׁי הָעִיר אֶל אֱלִישָׁע הִנֵּה נָא מוֹשַׁב הָעִיר טוֹב כַּאֲשֶׁר אֲדֹנִי רֹאֶה וְגוֹ׳״. וְכִי מֵאַחַר דְּמַיִם רָעִים וְאֶרֶץ מְשַׁכֶּלֶת, אֶלָּא מָה טוֹבָתָהּ? אָמַר רַבִּי חָנִין: חֵן מָקוֹם עַל יוֹשְׁבָיו. אָמַר רַבִּי יוֹחָנָן, שְׁלֹשָׁה חִינּוֹת הֵן: חֵן מָקוֹם עַל יוֹשְׁבָיו, חֵן אִשָּׁה עַל בַּעְלָהּ, חֵן מִקָּח עַל (מִקָּחוֹ).
אנשי יריחו אמרו לאלישע: מושב העיר טוב — אבל המים רעים! אם הכל רע, מה טוב? ר' חנין: חן מקום על יושביו. ר' יוחנן: שלושה חינות — ① חן מקום על יושביו ② חן אשה על בעלה ③ חן מקח על קונהו.
תָּנוּ רַבָּנַן, שְׁלֹשָׁה חֲלָאִין חָלָה אֱלִישָׁע: אֶחָד שֶׁגֵּירָה דּוּבִּים בַּתִּינוֹקוֹת, וְאֶחָד שֶׁדְּחָפוֹ לְגֵחֲזִי בִּשְׁתֵּי יָדַיִם, וְאֶחָד שֶׁמֵּת בּוֹ, שֶׁנֶּאֱמַר: ״וֶאֱלִישָׁע חָלָה אֶת חׇלְיוֹ אֲשֶׁר יָמוּת בּוֹ״.
אלישע חלה שלוש פעמים: ① על שגירה דובים בנערים. ② על שדחף את גיחזי בשתי ידיים ולא השאיר לו פתח לתשובה. ③ חולי שמת בו.
תָּנוּ רַבָּנַן: לְעוֹלָם תְּהֵא שְׂמֹאל דּוֹחָה וְיָמִין מְקָרֶבֶת, לֹא כֶּאֱלִישָׁע שֶׁדְּחָפוֹ לְגֵחֲזִי בִּשְׁתֵּי יָדָיו, וְלֹא כִּיהוֹשֻׁעַ בֶּן פְּרַחְיָה שֶׁדְּחָפוֹ לְיֵשׁוּ הַנּוֹצְרִי מִתַּלְמִידָיו בִּשְׁתֵּי יָדָיו.
כלל חשוב: שמאל דוחה וימין מקרבת — גם כשדוחים, יש להשאיר דרך חזרה. דוגמאות שליליות: ① אלישע דחף את גיחזי בשתי ידיים. ② יהושע בן פרחיה דחף את ישו הנוצרי בשתי ידיים.
אֱלִישָׁע מַאי הִיא, דִּכְתִיב: ״וַיֹּאמֶר נַעֲמָן הוֹאֵל קַח כִּכָּרָיִם״, וּכְתִיב: ״וַיֹּאמֶר אֵלָיו לֹא לִבִּי הָלַךְ כַּאֲשֶׁר הָפַךְ אִישׁ מֵעַל מֶרְכַּבְתּוֹ לִקְרָאתֶךָ הַעֵת לָקַחַת אֶת הַכֶּסֶף וְלָקַחַת בְּגָדִים וְזֵיתִים וּכְרָמִים וְצֹאן וּבָקָר וַעֲבָדִים וּשְׁפָחוֹת״.
מעשה גיחזי: נעמן נרפא מצרעתו, גיחזי לקח ממנו כסף ובגדים שלא ברשות אלישע. אלישע אמר לו: "לא ליבי הלך... העת לקחת כסף ובגדים וזיתים וכרמים?"
וּמִי שָׁקֵיל כּוּלֵּי הַאי? כֶּסֶף וּבְגָדִים הוּא דְּשָׁקֵיל! אָמַר רַבִּי יִצְחָק: בְּאוֹתָהּ שָׁעָה הָיָה אֱלִישָׁע עוֹסֵק בִּשְׁמֹנָה שְׁרָצִים, אָמַר לוֹ: רָשָׁע, הִגִּיעַ עֵת לִיטּוֹל שְׂכַר שְׁמֹנָה שְׁרָצִים. ״וְצָרַעַת נַעֲמָן תִּדְבַּק בְּךָ וּבְזַרְעֲךָ לְעוֹלָם״. ״וְאַרְבָּעָה אֲנָשִׁים הָיוּ מְצֹרָעִים״, אָמַר רַבִּי יוֹחָנָן: זֶה גֵּחֲזִי וּשְׁלֹשֶׁת בָּנָיו.
קושיה: הרי רק כסף ובגדים לקח! תשובה (ר' יצחק): אלישע למד אז סוגיית 8 שרצים. אמר לגיחזי: הגיע עתך ליטול שכר 8 השרצים (= 8 דברים). צרעת נעמן נדבקה בו ובזרעו. ארבעת המצורעים (מלכים ב) = גיחזי ושלושת בניו.
״וַיָּבֹא אֱלִישָׁע דַּמֶּשֶׂק״. לָמָּה הָלַךְ? אָמַר רַבִּי יוֹחָנָן: שֶׁהָלַךְ לְהַחְזִירוֹ לְגֵחֲזִי בִּתְשׁוּבָה, וְלֹא חָזַר. אָמַר לוֹ: חֲזוֹר בָּךְ. אָמַר לוֹ: כָּךְ מְקוּבְּלַנִי מִמְּךָ, כׇּל מִי שֶׁחָטָא וְהֶחְטִיא אֶת הָרַבִּים — אֵין מַסְפִּיקִין בְּיָדוֹ לַעֲשׂוֹת תְּשׁוּבָה.
אלישע הלך לדמשק כדי להחזיר את גיחזי בתשובה. גיחזי סירב: "מקובלני ממך — מי שחטא והחטיא את הרבים אין מספיקין בידו לעשות תשובה."
מַאי עֲבַד? אִיכָּא דְּאָמְרִי: אֶבֶן שׁוֹאֶבֶת תָּלָה לוֹ לְחַטַּאת יָרָבְעָם, וְהֶעֱמִידוֹ בֵּין שָׁמַיִם לָאָרֶץ. וְאִיכָּא דְּאָמְרִי: שֵׁם חֲקַק לַהּ אַפּוּמַּהּ, וְהָיְתָה אוֹמֶרֶת ״אָנֹכִי״ וְ״לֹא יִהְיֶה לְךָ״.
מה עשה גיחזי? ① אבן שואבת (מגנט) תלה לעגל ירבעם והעמידו באוויר — והיו עובדים אותו. ② שם קדוש חקק על פי העגל שהיה אומר "אנכי" ו"לא יהיה לך". ③ דחה חכמים מלפני אלישע — שלא יבואו ללמוד.
וְאִיכָּא דְּאָמְרִי: רַבָּנַן דְּחָה מִקַּמֵּיהּ, דִּכְתִיב: ״וַיֹּאמְרוּ בְנֵי הַנְּבִיאִים אֶל אֱלִישָׁע הִנֵּה נָא הַמָּקוֹם אֲשֶׁר אֲנַחְנוּ יֹשְׁבִים שָׁם לְפָנֶיךָ צַר מִמֶּנּוּ״, מִכְּלָל דְּעַד הָאִידָּנָא לָא הֲוָה דְּחִיק.
ראיה לדחיית החכמים: "הנה נא המקום... צר ממנו" — עד עכשיו לא היה צר, כי גיחזי היה מרחיק את התלמידים.
יְהוֹשֻׁעַ בֶּן פְּרַחְיָה מַאי הִיא — כְּדַהֲוָה קָא קָטֵיל יַנַּאי מַלְכָּא לְרַבָּנַן, שִׁמְעוֹן בֶּן שָׁטַח (אַטְמִינֵהוּ) [אַטְמַרְתֵּיהּ] אֲחָתֵיהּ, רַבִּי יְהוֹשֻׁעַ בֶּן פְּרַחְיָה אֲזַל עֲרַק לַאֲלֶכְּסַנְדְּרִיָּא שֶׁל מִצְרַיִם. כִּי הֲוָה שְׁלָמָא, שְׁלַח לֵיהּ שִׁמְעוֹן בֶּן שָׁטַח: מִנַּי יְרוּשָׁלַיִם עִיר הַקּוֹדֶשׁ לִךְ אֲלֶכְּסַנְדְּרִיָּא שֶׁל מִצְרַיִם: אֲחוֹתִי! בַּעֲלִי שָׁרוּי בְּתוֹכֵךְ, וַאֲנִי יוֹשֶׁבֶת שׁוֹמֵמָה! אֲמַר: שְׁמַע מִינַּהּ הֲוָה לֵיהּ שְׁלָמָא.
כשינאי המלך הרג את החכמים — שמעון בן שטח הוטמן על ידי אחותו (אשת ינאי), ויהושע בן פרחיה ברח למצרים. כשנעשה שלום, שמעון שלח לו מכתב מרומז: "אחותי, בעלי בתוכך ואני יושבת שוממה" = ירושלים מחכה לך.
כִּי אֲתָא, אִקְּלַע לְהָהוּא אוּשְׁפִּיזָא, קָם קַמַּיְיהוּ בִּיקָרָא שַׁפִּיר, עָבְדִי לֵיהּ יְקָרָא טוּבָא, יָתֵיב וְקָא מִשְׁתַּבַּח: כַּמָּה נָאָה אַכְסַנְיָא זוֹ! אֲמַר לֵיהּ יֵשׁוּ הַנּוֹצְרִי: רַבִּי, עֵינֶיהָ טְרוּטוֹת. אֲמַר לֵיהּ: רָשָׁע, בְּכָךְ אַתָּה עוֹסֵק? אַפֵּיק אַרְבַּע מְאָה שִׁפּוּרֵי וְשַׁמְּתֵיהּ. כׇּל יוֹמָא אֲתָא לְקַמֵּיהּ וְלָא קַבְּלֵיהּ.
בחזרתו, התארח בפונדק שכיבדו אותו מאוד. ר' יהושע שיבח את האכסניה. תלמידו (ישו) אמר: "רבי, עיניה טרוטות" (= לקוי מראה). ר' יהושע כעס: רשע, בכך אתה עוסק?! נידה אותו ב-400 שופרות.
יוֹמָא חַד הֲוָה קָרֵי קְרִיַּת שְׁמַע, אֲתָא לְקַמֵּיהּ. הֲוָה בְּדַעְתֵּיהּ לְקַבּוֹלֵיהּ, אַחְוִי לֵיהּ בִּידֵיהּ, סְבַר מִדְחָא דָּחֵי לֵיהּ, אֲזַל זְקַף לְבֵינְתָּא פַּלְחַאּ. אֲמַר לֵיהּ: חֲזוֹר בָּךְ! אֲמַר לֵיהּ: כָּךְ מְקוּבְּלַנִי מִמְּךָ, כׇּל הַחוֹטֵא וּמַחְטִיא אֶת הָרַבִּים — אֵין מַסְפִּיקִין בְּיָדוֹ לַעֲשׂוֹת תְּשׁוּבָה. דְּאָמַר מָר: יֵשׁוּ הַנּוֹצְרִי כִּישֵּׁף וְהִסִּית וְהִדִּיחַ וְהֶחְטִיא אֶת יִשְׂרָאֵל.
כל יום בא התלמיד לפניו ולא קיבלו. יום אחד, בזמן קריאת שמע, רצה לקבלו — הראה לו בידו שימתין. התלמיד חשב שדוחה אותו לגמרי, הלך והעמיד לבנה ועבד אותה. ר' יהושע אמר: חזור! ענה: מקובלני ממך — החוטא והמחטיא את הרבים אין מספיקין בידו לעשות תשובה.
תַּנְיָא, רַבִּי שִׁמְעוֹן בֶּן אֶלְעָזָר אוֹמֵר: יֵצֶר, תִּינוֹק, וְאִשָּׁה — תְּהֵא שְׂמֹאל דּוֹחָה וְיָמִין מְקָרֶבֶת.
ר' שמעון בן אלעזר: יצר, תינוק ואשה — שמאל דוחה וימין מקרבת. הכלל: יש לשמור על האיזון — לא לדחות יותר מדי.
מַתְנִי׳ נִמְצָא הַהוֹרֵג עַד שֶׁלֹּא נֶעֶרְפָה הָעֶגְלָה — תֵּצֵא וְתִרְעֶה בָּעֵדֶר. מִשֶּׁנֶּעֶרְפָה הָעֶגְלָה — תִּקָּבֵר בִּמְקוֹמָהּ, שֶׁעַל סָפֵק בָּאתָה מִתְּחִילָּתָהּ — כִּיפְּרָה סְפֵיקָהּ, וְהָלְכָה לָהּ. נֶעֶרְפָה הָעֶגְלָה וְאַחַר כָּךְ נִמְצָא הַהוֹרֵג — הֲרֵי זֶה יֵהָרֵג.
דין העגלה כשנמצא הרוצח: ① לפני העריפה — העגלה חוזרת לרעות בעדר (אין בה קדושה). ② אחרי העריפה — תיקבר במקומה, כי באה על ספק וכיפרה ספיקה. ③ אם נערפה ואחר כך נמצא הרוצח — הרוצח נהרג ואין אומרים שכבר נתכפר.
עֵד אֶחָד אוֹמֵר ״רָאִיתִי אֶת הַהוֹרֵג״, וְעֵד אֶחָד אוֹמֵר ״לֹא רָאִיתָ״. אִשָּׁה אוֹמֶרֶת ״רָאִיתִי״, וְאִשָּׁה אוֹמֶרֶת ״לֹא רָאִית״ — הָיוּ עוֹרְפִין. עֵד אֶחָד אוֹמֵר ״רָאִיתִי״, וּשְׁנַיִם אוֹמְרִים ״לֹא רָאִיתָ״ — הָיוּ עוֹרְפִין. שְׁנַיִם אוֹמְרִים ״רָאִינוּ״, וְאֶחָד אוֹמֵר לָהֶן ״לֹא רְאִיתֶם״ — לֹא הָיוּ עוֹרְפִין.
דיני עדות: ① עד אחד אומר "ראיתי" ועד אחד מכחיש — עורפים (אין עדות ברורה). ② עד אחד אומר "ראיתי" ושניים אומרים "לא ראית" — עורפים. ③ שניים אומרים "ראינו" ואחד מכחיש — לא עורפים.
מִשֶּׁרַבּוּ הָרוֹצְחָנִין — בָּטְלָה עֶגְלָה עֲרוּפָה. מִשֶּׁבָּא אֱלִיעֶזֶר בֶּן דִּינַאי, וּתְחִינָּה בֶּן פְּרִישָׁה הָיָה נִקְרָא, חָזְרוּ לִקְרוֹתוֹ ״בֶּן הָרַצְחָן״.
משרבו הרוצחים — בטלה עגלה ערופה. משבא אליעזר בן דינאי (שנקרא "תחינה בן פרישה" ואחר כך "בן הרצחן") — בטלו העגלות, כי ידעו מי הרוצח.
מִשֶּׁרַבּוּ הַמְנָאֲפִים — פָּסְקוּ הַמַּיִם הַמָּרִים. וְרַבִּי יוֹחָנָן בֶּן זַכַּאי הִפְסִיקָן, שֶׁנֶּאֱמַר: ״לֹא אֶפְקוֹד עַל בְּנוֹתֵיכֶם כִּי תִזְנֶינָה וְעַל כַּלּוֹתֵיכֶם כִּי תְנָאַפְנָה כִּי הֵם וְגוֹ׳״.
משרבו המנאפים — פסקו המים המרים (מי סוטה). ר' יוחנן בן זכאי הפסיקם, שנאמר: "לא אפקוד על בנותיכם כי תזנינה" — כשהאבות עצמם חוטאים, המים אינם בודקים.
מִשֶּׁמֵּת יוֹסֵי בֶּן יוֹעֶזֶר אִישׁ צְרֵידָה וְיוֹסֵי בֶּן יְהוּדָה אִישׁ יְרוּשָׁלַיִם בָּטְלוּ הָאֶשְׁכֹּלוֹת, שֶׁנֶּאֱמַר: ״אֵין אֶשְׁכּוֹל לֶאֱכוֹל בִּכּוּרָה אִוְּתָה נַפְשִׁי״, יוֹחָנָן כֹּהֵן גָּדוֹל הֶעֱבִיר הוֹדָיַית הַמַּעֲשֵׂר. אַף הוּא בִּטֵּל אֶת הַמְעוֹרְרִין וְאֶת הַנּוֹקְפִין.
משמת יוסי בן יועזר ויוסי בן יהודה — בטלו ה"אשכולות" (= אנשים שהכל בהם). שנאמר: "אין אשכול לאכול."
עַד יָמָיו הָיָה פַּטִּישׁ מַכֶּה בִּירוּשָׁלַיִם. וּבְיָמָיו אֵין צָרִיךְ לִשְׁאוֹל עַל הַדְּמַאי.
"עד ימיו היה פטיש מכה בירושלים" = מלאכה בחול המועד. יוחנן כהן גדול ביטל זאת. בימיו אין צריך לשאול על הדמאי — כי הכל עישרו.
גְּמָ׳ תָּנוּ רַבָּנַן: מִנַּיִן שֶׁאִם נֶעֶרְפָה הָעֶגְלָה וְאַחַר כָּךְ נִמְצָא הַהוֹרֵג שֶׁאֵין פּוֹטֶרֶת אוֹתוֹ — תַּלְמוּד לוֹמַר: ״וְלָאָרֶץ לֹא יְכֻפַּר לַדָּם אֲשֶׁר שֻׁפַּךְ בָּהּ כִּי אִם בְּדַם שֹׁפְכוֹ״.
ברייתא: נערפה העגלה ונמצא הרוצח — אין פוטרת אותו. מקור: "ולארץ לא יכופר לדם אשר שופך בה כי אם בדם שופכו" — רק עונש הרוצח מכפר באמת.
עֵד אֶחָד אוֹמֵר רָאִיתִי אֶת הַהוֹרֵג כּוּ׳. טַעְמָא דְּמַכְחֵישׁ לֵיהּ, הָא לָא מַכְחֵישׁ לֵיהּ — עֵד אֶחָד מְהֵימַן.
דיוק מהמשנה: "עד אחד אומר ראיתי — ומכחישו אחד" = עורפים. מכלל — אם לא מכחישו, עד אחד נאמן ולא עורפים.
מְנָהָנֵי מִילֵּי? דְּתָנוּ רַבָּנַן: ״לֹא נוֹדַע מִי הִכָּהוּ״. הָא נוֹדַע מִי הִכָּהוּ, אֲפִילּוּ אֶחָד בְּסוֹף הָעוֹלָם — לֹא הָיוּ עוֹרְפִין. רַבִּי עֲקִיבָא אוֹמֵר: מִנַּיִן לְסַנְהֶדְרִין שֶׁרָאוּ אֶחָד שֶׁהָרַג אֶת הַנֶּפֶשׁ וְאֵין מַכִּירִין אוֹתוֹ, שֶׁלֹּא הָיוּ עוֹרְפִין — תַּלְמוּד לוֹמַר: ״וְעֵינֵינוּ לֹא רָאוּ״, וַהֲלֹא רָאוּ.
מקור: "לא נודע מי הכהו" — אם נודע, אפילו אחד בסוף העולם — לא עורפים. ר' עקיבא: סנהדרין שראו את הרוצח אך לא מזהים אותו — לא עורפים.
הַשְׁתָּא דְּאָמְרַתְּ עֵד אֶחָד מְהֵימַן, אִידַּךְ חַד הֵיכִי מָצֵי מַכְחֵישׁ לֵיהּ? וְהָאָמַר עוּלָּא: כׇּל מָקוֹם שֶׁהֶאֱמִינָה תּוֹרָה עֵד אֶחָד — הֲרֵי כָּאן שְׁנַיִם, וְאֵין דְּבָרָיו שֶׁל אֶחָד בִּמְקוֹם שְׁנַיִם! אָמַר לָךְ עוּלָּא, תְּנִי: ״לֹא הָיוּ עוֹרְפִין״. וְכֵן אָמַר רַבִּי יִצְחָק, תְּנִי: ״לֹא הָיוּ עוֹרְפִין״.
קושיה: אם עד אחד נאמן כשניים, איך אחד מכחישו? עולא: כל מקום שהתורה האמינה עד אחד = כשניים! תשובה: עולא ור' יצחק גורסים "לא היו עורפים" (לא כגרסת המשנה שלפנינו).
וְרַבִּי חִיָּיא אָמַר, תְּנִי: ״הָיוּ עוֹרְפִין״. וּלְרַבִּי חִיָּיא קַשְׁיָא דְּעוּלָּא! לָא קַשְׁיָא: כָּאן בְּבַת אַחַת, כָּאן בְּזֶה אַחַר זֶה.
ר' חייא גורס: "היו עורפים" (כגרסתנו). קושיה מדברי עולא? תשובה: כשבאו בבת אחת — מבטלים זה את זה. כשבאו בזה אחר זה — הראשון קיים.
תְּנַן: עֵד אֶחָד אוֹמֵר ״רָאִיתִי אֶת הַהוֹרֵג״, וּשְׁנַיִם אוֹמְרִים ״לֹא רָאִיתָ״ — הָיוּ עוֹרְפִין. הָא חַד וְחַד — לֹא הָיוּ עוֹרְפִין, תְּיוּבְתָּא דְּרַבִּי חִיָּיא!
דיוק: "עד אחד ושניים מכחישים — עורפים". מכלל: אחד ואחד — לא עורפים? זו קושיה על ר' חייא!
וְלִיטַעְמָיךְ, אֵימָא סֵיפָא: שְׁנַיִם אוֹמְרִים ״רָאִינוּ״, וְעֵד אֶחָד אוֹמֵר ״לֹא רְאִיתֶם״ — לֹא הָיוּ עוֹרְפִין. הָא חַד וְחַד — הָיוּ עוֹרְפִין.
דחייה: גם מהסיפא אפשר לדייק הפוך. אלא: כל המשנה עוסקת בפסולי עדות (כנשים), וכשיטת ר' נחמיה — הולכים אחר רוב דעות בפסולי עדות.
אֶלָּא: מַתְנִיתִין כּוּלַּהּ בִּפְסוּלֵי עֵדוּת, וְכִדְרַבִּי נְחֶמְיָה, דְּאָמַר: כׇּל מָקוֹם שֶׁהֶאֱמִינָה תּוֹרָה עֵד אֶחָד הַלֵּךְ אַחַר רוֹב דֵּעוֹת, וְעָשׂוּ שְׁתֵּי נָשִׁים בְּאִישׁ אֶחָד כִּשְׁנֵי אֲנָשִׁים בְּאִישׁ אֶחָד.
ר' נחמיה: כל מקום שעד אחד נאמן — הולכים אחר רוב דעות. שתי נשים נגד איש אחד = כשני אנשים נגד אחד.
וְאִיכָּא דְאָמְרִי: כׇּל הֵיכָא דַּאֲתָא עֵד אֶחָד — כָּשֵׁר מֵעִיקָּרָא, אֲפִילּוּ מֵאָה נָשִׁים כִּי אֶחָד דָּמְיָין. וְהָכָא בְּמַאי עָסְקִינַן — כְּגוֹן דַּאֲתַאי אִשָּׁה מֵעִיקָּרָא. וְתָרְצַהּ לִדְרַבִּי נְחֶמְיָה הָכִי: רַבִּי נְחֶמְיָה אוֹמֵר: כׇּל מָקוֹם שֶׁהֶאֱמִינָה תּוֹרָה עֵד אֶחָד — הַלֵּךְ אַחַר רוֹב דֵּעוֹת, וְעָשׂוּ שְׁתֵּי נָשִׁים בְּאִשָּׁה אַחַת כִּשְׁנֵי אֲנָשִׁים בְּאִישׁ אֶחָד, אֲבָל שְׁתֵּי נָשִׁים בְּאִישׁ אֶחָד — כִּי פַּלְגָא וּפַלְגָא דָּמֵי.
לשון אחר: עד כשר מתחילה — אפילו מאה נשים מכחישות = כעד אחד. המשנה עוסקת כשהעדה הראשונה היא אשה — ואז שתי נשים מכריעות כנגדה.
וְתַרְתֵּי פְּסוּלֵי עֵדוּת, לְמָה לִי? מַהוּ דְּתֵימָא: כִּי אָזְלִינַן בָּתַר רוֹב דֵּעוֹת — לְחוּמְרָא, אֲבָל לְקוּלָּא — לָא, קָא מַשְׁמַע לַן.
"שתי נשים באישה אחת = כשני אנשים באיש" — אבל שתי נשים באיש כשר = חצי וחצי.
מִשֶּׁרַבּוּ הָרוֹצְחִין כּוּ׳. תָּנוּ רַבָּנַן: מִשֶּׁרַבּוּ הָרוֹצְחָנִין — בָּטְלָה עֶגְלָה עֲרוּפָה, לְפִי שֶׁאֵינָהּ בָּאָה אֶלָּא עַל הַסָּפֵק. מִשֶּׁרַבּוּ הָרוֹצְחָנִין בְּגָלוּי — בָּטְלָה עֶגְלָה עֲרוּפָה.
שאלה: למה צריך שני חלקים במשנה (רישא וסיפא) — שניהם מלמדים אותו כלל? תשובה: שלא תאמר שהולכים אחר רוב רק לחומרא (= להביא עגלה). באה הסיפא ללמד: גם לקולא (= לא להביא).
מִשֶּׁרַבּוּ הַנּוֹאֲפִין כּוּ׳. תָּנוּ רַבָּנַן: ״וְנִקָּה הָאִישׁ מֵעָוֹן״, בִּזְמַן שֶׁהָאִישׁ מְנוּקֶּה מֵעָוֹן — הַמַּיִם בּוֹדְקִין אֶת אִשְׁתּוֹ, אֵין הָאִישׁ מְנוּקֶּה מֵעָוֹן — אֵין הַמַּיִם בּוֹדְקִין אֶת אִשְׁתּוֹ. וְאוֹמֵר: ״לֹא אֶפְקוֹד עַל בְּנוֹתֵיכֶם כִּי תִזְנֶינָה כּוּ׳״.
משרבו הרוצחים — בטלה עגלה ערופה. ברייתא: כי העגלה באה רק על ספק. משרבו הרוצחים בגלוי — אין ספק.
מַאי ״וְאוֹמֵר״? וְכִי תֵּימָא עָוֹן דִּידֵיהּ — אִין, דִּבְנֵיהּ וְדִבְנָתֵיהּ — לָא, תָּא שְׁמַע: ״לֹא אֶפְקוֹד עַל בְּנוֹתֵיכֶם כִּי תִזְנֶינָה וְעַל כַּלּוֹתֵיכֶם כִּי תְנָאַפְנָה״.
משרבו הנואפים — בטלו מי סוטה. ברייתא: "ונקה האיש מעוון" = רק כשהבעל מנוקה מעוון — המים בודקים. אם הבעל עצמו חוטא — המים לא פועלים.
וְכִי תֵּימָא: עֲוֹן אֵשֶׁת אִישׁ — אִין, עָוֹן דִּפְנוּיָה — לָא, תָּא שְׁמַע: ״כִּי הֵם עִם הַזֹּנוֹת יְפָרֵדוּ וְעִם הַקְּדֵשׁוֹת יְזַבֵּחוּ וְגוֹ׳״.
שלוש דרשות מפסוקי הושע: ① עוון הבעל עצמו — מונע את בדיקת המים. ② עוון בניו ובנותיו — גם מונע. ③ לא רק ניאוף אלא גם בעילת פנויה — "כי הם עם הזונות יפרדו."
מַאי ״וְעָם לֹא יָבִין יִלָּבֵט״? אָמַר רַבִּי אֶלְעָזָר: אָמַר לָהֶם נָבִיא לְיִשְׂרָאֵל: אִם אַתֶּם מַקְפִּידִין עַל עַצְמְכֶם — מַיִם בּוֹדְקִין נְשׁוֹתֵיכֶם, וְאִם לָאו — אֵין הַמַּיִם בּוֹדְקִין נְשׁוֹתֵיכֶם.
"ועם לא יבין ילבט" = ר' אלעזר: הנביא אמר לישראל — אם אתם מקפידים על עצמכם, המים בודקים. אם לאו — אין המים בודקים, ואתם נשארים מתלבטים בספק.
מִשֶּׁרַבּוּ בַּעֲלֵי הֲנָאָה, נִתְעַוְּתוּ הַדִּינִין וְנִתְקַלְקְלוּ הַמַּעֲשִׂים וְאֵין נוֹחַ בָּעוֹלָם. מִשֶּׁרַבּוּ רוֹאֵי פָנִים בַּדִּין, בָּטַל ״לֹא תָגוּרוּ״, וּפָסַק ״לֹא תַכִּירוּ״, וּפָרְקוּ עוֹל שָׁמַיִם וְנָתְנוּ עֲלֵיהֶם עוֹל בָּשָׂר וָדָם.
משרבו בעלי הנאה — נתעוותו הדינים ונתקלקלו המעשים. משרבו רואי פנים בדין — בטל "לא תגורו" ו"לא תכירו פנים", פרקו עול שמים ונתנו עול בשר ודם.
מִשֶּׁרַבּוּ לוֹחֲשֵׁי לְחִישׁוֹת בַּדִּין — רָבָה חֲרוֹן אַף בְּיִשְׂרָאֵל, וְנִסְתַּלְּקָה הַשְּׁכִינָה, מִשּׁוּם שֶׁנֶּאֱמַר: ״בְּקֶרֶב אֱלֹהִים יִשְׁפֹּט״. מִשֶּׁרַבּוּ ״אַחֲרֵי בִצְעָם לִבָּם הֹלֵךְ״ — רַבּוּ ״הָאוֹמְרִים לָרַע טוֹב וְלַטּוֹב רָע״. מִשֶּׁרַבּוּ ״הָאוֹמְרִים לָרַע טוֹב וְלַטּוֹב רָע״ — רַבּוּ ״הוֹי הוֹי״ בָּעוֹלָם.
משרבו לוחשי לחישות בדין — רבה חרון אף ונסתלקה שכינה. משרבו הרודפים אחר בצע — רבו "האומרים לרע טוב". ומזה — רבו "הוי" בעולם.
מִשֶּׁרַבּוּ מוֹשְׁכֵי הָרוֹק — רַבּוּ הַיְּהִירִים וְנִתְמַעֲטוּ הַתַּלְמִידִים, וְהַתּוֹרָה חוֹזֶרֶת עַל לוֹמְדֶיהָ. מִשֶּׁרַבּוּ הַיְּהִירִים — הִתְחִילוּ בְּנוֹת יִשְׂרָאֵל לְהִנָּשֵׂא לִיהִירִים, שֶׁאֵין דּוֹרֵינוּ רוֹאֶה אֶלָּא לַפָּנִים.
משרבו מושכי הרוק (גאוותנים) — רבו יהירים ונתמעטו תלמידים, והתורה חוזרת על לומדיה. משרבו יהירים — בנות ישראל נישאות להם, כי הדור רואה רק חיצוניות.
אִינִי?! וְהָאָמַר מָר: הַאי מַאן דְּמִיַּהַר — אֲפִילּוּ אַאִינָשֵׁי בֵּיתֵיהּ לָא מִיקַּבַּל, שֶׁנֶּאֱמַר: ״גֶּבֶר יָהִיר וְלֹא יִנְוֶה״, לֹא יִנְוֶה אֲפִילּוּ בַּנָּוֶה שֶׁלּוֹ! מֵעִיקָּרָא קָפְצָה עֲלֵיהּ, לְסוֹף מִיתְּזִיל עֲלַיְיהוּ.
קושיה: הרי יהיר — אפילו אשתו אינה סובלת אותו! "גבר יהיר ולא ינוה" = אפילו בביתו אינו מתקבל! תשובה: בהתחלה קופצת עליו (נמשכת), אבל בסוף מזלזלת בו.
מִשֶּׁרַבּוּ מַטִּילֵי מְלַאי עַל בַּעֲלֵי בָתִּים — רָבָה הַשּׁוֹחַד וְהַטָּיַית מִשְׁפָּט, וּפָסְקָה טוֹבָה. מִשֶּׁרַבּוּ ״מְקַבְּלֵנִי טוֹבָתְךָ״ וּ״מְחַזְּקֵנִי טוֹבוֹתֶיךָ״ — רַבּוּ ״אִישׁ הַיָּשָׁר בְּעֵינָיו יַעֲשֶׂה״. שְׁפָלִים הוּגְבְּהוּ, וְהַגְּבוֹהִים הוּשְׁפְּלוּ, וּמַלְכוּתָא אָזְלָא וְנָוְלָא. מִשֶּׁרַבּוּ צָרֵי עַיִן וְטוֹרְפֵי טֶרֶף — רַבּוּ מְאַמְּצֵי הַלֵּב וְקוֹפְצֵי יָדַיִם מִלְּהַלְווֹת, וְעָבְרוּ עַל מַה שֶּׁכָּתוּב בַּתּוֹרָה ״הִשָּׁמֶר לְךָ פֶּן וְגוֹ׳״.
משרבו מטילי מלאי (= הוצאות) על בעלי בתים — רבה שוחד ועיוות דין. משרבו "מקבלני טובתך" — שפלים הוגבהו וגבוהים הושפלו, והמלכות מתנוולת.
מִשֶּׁרַבּוּ ״נְטוּיוֹת גָּרוֹן וּמְשַׂקְּרוֹת עֵינָיִם״ — רַבּוּ מַיִם הַמָּרִים, אֶלָּא שֶׁפָּסְקוּ. מִשֶּׁרַבּוּ מְקַבְּלֵי מַתָּנוֹת — נִתְמַעֲטוּ הַיָּמִים וְנִתְקַצְּרוּ הַשָּׁנִים, דִּכְתִיב: ״וְשׂוֹנֵא מַתָּנֹת יִחְיֶה״. מִשֶּׁרַבּוּ זְחוּחֵי הַלֵּב — רַבּוּ מַחֲלוֹקֹת בְּיִשְׂרָאֵל. מִשֶּׁרַבּוּ תַּלְמִידֵי שַׁמַּאי וְהִילֵּל שֶׁלֹּא שִׁימְּשׁוּ כׇּל צוֹרְכָּן — רַבּוּ מַחְלוֹקוֹת בְּיִשְׂרָאֵל, וְנַעֲשֵׂית תּוֹרָה כִּשְׁתֵּי תוֹרוֹת. מִשֶּׁרַבּוּ מְקַבְּלֵי צְדָקָה מִן הַנׇּכְרִי — הָיוּ יִשְׂרָאֵל לְמַעְלָה וְהֵם לְמַטָּה, יִשְׂרָאֵל לִפְנִים וְהֵם לְאָחוֹר.
משרבו צרי עין — רבו מאמצי הלב וקופצי ידים מלהלוות, ועברו על "השמר לך פן יהיה דבר עם לבבך" ו"לא תקפוץ את ידך."
מִשֶּׁמֵּת יוֹסֵי בֶּן יוֹעֶזֶר כּוּ׳. מַאי ״אֶשְׁכּוֹלוֹת״? אָמַר רַב יְהוּדָה אָמַר שְׁמוּאֵל, אִישׁ שֶׁהַכֹּל בּוֹ. יוֹחָנָן כֹּהֵן גָּדוֹל הֶעֱבִיר הוֹדָיַית הַמַּעֲשֵׂר כּוּ׳. מַאי טַעְמָא? אָמַר רַבִּי יוֹסֵי בְּרַבִּי חֲנִינָא: לְפִי שֶׁאֵין נוֹתְנִין אוֹתוֹ כְּתִיקּוּנוֹ. דְּרַחֲמָנָא אָמַר דְּיָהֲבִי לִלְוִיִּם,
משרבו "נטויות גרון" (נשים פרוצות) — רבו מי סוטה, אלא שפסקו כי רב הזנות. משרבו מקבלי מתנות — נתקצרו ימים ושנים. משרבו זחוחי הלב — רבו מחלוקות. משרבו תלמידי שמאי והלל שלא שימשו כל צרכם — נעשית תורה כשתי תורות.
